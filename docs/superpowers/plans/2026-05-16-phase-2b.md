# Phase 2b Implementation Plan — SAM 3 + DB-Driven Prompt + Ablation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Swap GroundingDINO + InsectSAM for SAM 3 as both detector and segmenter, ship the DB-driven prompt builder with version hashing, build the IoU-based label transfer tool with sanity-check validation, and run the ablation test (with-mask vs without-mask classifier AUC) to decide whether the segmenter earns its keep.

**Architecture:** SAM 3 is added to the `detectors/` and `segmenters/` factory registries. The same SAM 3 model instance powers both — bbox-prompted segmentation reuses the model weights from the text-prompted detection call (per parent spec §91). After SAM 3 ships, the user runs IoU-based label transfer on the existing labels, validates with a 20-image sanity check, and proceeds to the bbox-only labeling marathon under the new variant.

**Tech Stack:** Python 3.12, transformers 5.8.1 (SAM 3 in `Sam3Model` / `Sam3Processor`), PyTorch 2.12 on MPS (F32 dtype), polars 1.40, pyarrow 24.0, scikit-learn 1.5 (LogisticRegression for ablation).

```python
# scripts/detect_subjects/detectors/sam3.py
"""SAM 3 detector wrapper — text-prompted detection via facebook/sam3.

SAM 3 acts as BOTH detector and segmenter. This module covers the detection
half: text prompt -> bbox + per-instance metadata. The segmentation half is
in segmenters/sam3.py and shares the model via _sam3_shared.py.

_PER_INSTANCE_PHRASE_KEY: set to the key name found in the T0 probe (e.g.
"labels") if SAM 3 exposes per-instance phrase labels in
post_process_instance_segmentation results. Set to None to fall back to N
single-class calls (one per phrase).
"""
from __future__ import annotations
import hashlib
import json
import time
from typing import Optional

import torch
from PIL import Image
from transformers import Sam3Model, Sam3Processor

from scripts.detect_subjects.config import (
    BBOX_CONF_TOLERANCE,
    BBOX_MAX_AREA_RATIO,
    CACHE_DIR,
    HIGH_CONF_THRESHOLD,
    NMS_IOU_THRESHOLD,
)
from scripts.detect_subjects.interfaces import DetectionResult
from scripts.detect_subjects.metrics import iou_xywh_normalized
from scripts.detect_subjects._sam3_shared import _get_shared_sam3

# ── Probe-time constant ─────────────────────────────────────────────────────
# Update to the key name found by T0 probe (e.g. "labels"), or leave None.
_PER_INSTANCE_PHRASE_KEY: Optional[str] = None

SAM3_MODEL_ID = "facebook/sam3"
_SAM3_DET_CACHE_DIR = CACHE_DIR / "raw_sam3_det"
_SAM3_DET_CACHE_DIR.mkdir(parents=True, exist_ok=True)


def _det_cache_key(prompt_phrases: list[str], model_id: str) -> str:
    payload = model_id + "|" + "|".join(sorted(prompt_phrases))
    return hashlib.sha1(payload.encode()).hexdigest()[:10]


def _nms(
    boxes_scores: list[tuple[float, float, float, float, float]],
    iou_threshold: float,
) -> list[tuple[float, float, float, float, float]]:
    """Greedy NMS on (x, y, w, h, score) tuples, sorted descending by score."""
    boxes_scores = sorted(boxes_scores, key=lambda r: r[4], reverse=True)
    kept: list[tuple[float, float, float, float, float]] = []
    for cand in boxes_scores:
        cx, cy, cw, ch, cs = cand
        if any(
            iou_xywh_normalized((cx, cy, cw, ch), (k[0], k[1], k[2], k[3]))
            > iou_threshold
            for k in kept
        ):
            continue
        kept.append(cand)
    return kept


def _bark_beetle_fix(
    kept: list[tuple[float, float, float, float, float]],
) -> tuple[float, float, float, float, float]:
    """Pick the primary bbox: largest among high-conf candidates, not whole-frame."""
    top_conf = kept[0][4]
    candidates = [
        k for k in kept
        if k[4] >= top_conf - BBOX_CONF_TOLERANCE
        and (k[2] * k[3]) <= BBOX_MAX_AREA_RATIO
    ]
    return max(candidates, key=lambda k: k[2] * k[3]) if candidates else kept[0]


def _compute_distinct_subjects(
    kept: list[tuple[float, float, float, float, float]],
    phrases: Optional[list[Optional[str]]],
) -> list[tuple[float, float, float, float, float, Optional[str]]]:
    """Cluster NMS-kept boxes into distinct instances; annotate with phrase."""
    confident = [
        k for k in kept
        if k[4] >= HIGH_CONF_THRESHOLD
        and 0.005 <= (k[2] * k[3]) <= BBOX_MAX_AREA_RATIO
    ]
    result: list[tuple[float, float, float, float, float, Optional[str]]] = []
    for i, c in enumerate(confident):
        cx_center = c[0] + c[2] / 2.0
        cy_center = c[1] + c[3] / 2.0
        inside_any = False
        for s in result:
            if (s[0] <= cx_center <= s[0] + s[2]
                    and s[1] <= cy_center <= s[1] + s[3]):
                inside_any = True
                break
        if not inside_any:
            phrase = phrases[i] if (phrases is not None and i < len(phrases)) else None
            result.append((c[0], c[1], c[2], c[3], c[4], phrase))
    return result


class Sam3Detector:
    model_id: str = SAM3_MODEL_ID

    def __init__(
        self,
        device: str = "mps",
        dtype: torch.dtype = torch.float32,
        prompt_phrases: Optional[list[str]] = None,
        **kwargs,
    ) -> None:
        self.device = device
        self.dtype = dtype
        self.prompt_phrases: list[str] = (
            prompt_phrases if prompt_phrases is not None else ["an insect"]
        )
        self.processor, self.model = _get_shared_sam3(device, dtype)

    def _boxes_and_phrases_from_outputs(
        self,
        outputs,
        image: Image.Image,
        threshold: float,
        mask_threshold: float,
    ) -> tuple[
        list[tuple[float, float, float, float, float]],
        Optional[list[Optional[str]]],
    ]:
        results = self.processor.post_process_instance_segmentation(
            outputs,
            threshold=threshold,
            mask_threshold=mask_threshold,
            target_sizes=[(image.height, image.width)],
        )[0]
        raw_boxes = results.get("boxes", torch.zeros(0, 4))
        raw_scores = results.get("scores", torch.zeros(0))
        phrases: Optional[list[Optional[str]]] = None
        if _PER_INSTANCE_PHRASE_KEY and _PER_INSTANCE_PHRASE_KEY in results:
            phrases = list(results[_PER_INSTANCE_PHRASE_KEY])
        normalized = []
        for (x1, y1, x2, y2), score in zip(raw_boxes.tolist(), raw_scores.tolist()):
            normalized.append((
                x1 / image.width,
                y1 / image.height,
                (x2 - x1) / image.width,
                (y2 - y1) / image.height,
                float(score),
            ))
        return normalized, phrases

    def _single_class_inference(
        self,
        image: Image.Image,
        image_id: Optional[str],
        threshold: float,
        mask_threshold: float,
    ) -> tuple[
        list[tuple[float, float, float, float, float]],
        Optional[list[Optional[str]]],
    ]:
        all_boxes: list[tuple[float, float, float, float, float]] = []
        all_phrases: list[Optional[str]] = []
        for phrase in self.prompt_phrases:
            cache_key = _det_cache_key([phrase], SAM3_MODEL_ID)
            cache_path = (
                _SAM3_DET_CACHE_DIR / f"{image_id}__{cache_key}.json"
                if image_id else None
            )
            cached = None
            if cache_path is not None and cache_path.exists():
                try:
                    cached = json.loads(cache_path.read_text())
                except Exception:
                    cached = None
            if cached is not None:
                boxes_raw = cached["boxes"]
                scores_raw = cached["scores"]
            else:
                inputs = self.processor(
                    images=image, text=[phrase], return_tensors="pt"
                ).to(self.device)
                if "pixel_values" in inputs:
                    inputs["pixel_values"] = inputs["pixel_values"].to(self.dtype)
                with torch.no_grad():
                    out = self.model(**inputs)
                results = self.processor.post_process_instance_segmentation(
                    out,
                    threshold=threshold,
                    mask_threshold=mask_threshold,
                    target_sizes=[(image.height, image.width)],
                )[0]
                boxes_raw = results.get("boxes", torch.zeros(0, 4)).tolist()
                scores_raw = results.get("scores", torch.zeros(0)).tolist()
                if cache_path is not None:
                    try:
                        cache_path.write_text(
                            json.dumps({"boxes": boxes_raw, "scores": scores_raw})
                        )
                    except Exception:
                        pass
            for (x1, y1, x2, y2), score in zip(boxes_raw, scores_raw):
                all_boxes.append((
                    x1 / image.width,
                    y1 / image.height,
                    (x2 - x1) / image.width,
                    (y2 - y1) / image.height,
                    float(score),
                ))
                all_phrases.append(phrase)
        return all_boxes, all_phrases if all_boxes else None

    @torch.no_grad()
    def detect(
        self,
        image: Image.Image,
        image_id: Optional[str] = None,
    ) -> DetectionResult:
        from scripts.detect_subjects.config import SAM3_BOX_THRESHOLD, SAM3_MASK_THRESHOLD
        start = time.perf_counter()

        if _PER_INSTANCE_PHRASE_KEY is not None:
            cache_key = _det_cache_key(self.prompt_phrases, SAM3_MODEL_ID)
            cache_path = (
                _SAM3_DET_CACHE_DIR / f"{image_id}__{cache_key}.json"
                if image_id else None

            cached = None
            if cache_path is not None and cache_path.exists():
                try:
                    cached = json.loads(cache_path.read_text())
                except Exception:
                    cached = None
            if cached is not None:
                raw_normalized = [tuple(b) for b in cached["boxes_scores"]]
                raw_phrases: Optional[list[Optional[str]]] = cached.get("phrases")
            else:
                inputs = self.processor(
                    images=image,
                    text=self.prompt_phrases,
                    return_tensors="pt",
                ).to(self.device)
                if "pixel_values" in inputs:
                    inputs["pixel_values"] = inputs["pixel_values"].to(self.dtype)
                outputs = self.model(**inputs)
                raw_normalized, raw_phrases = self._boxes_and_phrases_from_outputs(
                    outputs, image,
                    threshold=SAM3_BOX_THRESHOLD,
                    mask_threshold=SAM3_MASK_THRESHOLD,
                )
                if cache_path is not None:
                    try:
                        cache_path.write_text(json.dumps({
                            "boxes_scores": [list(b) for b in raw_normalized],
                            "phrases": raw_phrases,
                        }))
                    except Exception:
                        pass
        else:
            raw_normalized, raw_phrases = self._single_class_inference(
                image, image_id,
                threshold=SAM3_BOX_THRESHOLD,
                mask_threshold=SAM3_MASK_THRESHOLD,
            )

        elapsed_ms = int((time.perf_counter() - start) * 1000)
        n_raw = len(raw_normalized)

        if not raw_normalized:
            return DetectionResult(
                bbox_xywh_normalized=None,
                confidence=None,
                n_raw_detections=0,
                n_distinct_detections=0,
                detection_ms=elapsed_ms,
                distinct_subjects=[],
            )

        kept = _nms(list(raw_normalized), NMS_IOU_THRESHOLD)

        # Build phrase map: kept-box -> phrase (phrases are parallel to raw_normalized)
        if raw_phrases is not None:
            raw_list = list(raw_normalized)
            phrase_by_box = {}
            for box in kept:
                try:
                    idx = raw_list.index(box)
                    phrase_by_box[id(box)] = raw_phrases[idx]
                except (ValueError, IndexError):
                    phrase_by_box[id(box)] = None
        else:
            phrase_by_box = {}

        top = _bark_beetle_fix(kept)
        top_phrase: Optional[str] = phrase_by_box.get(id(top))
        top_score: Optional[float] = top[4] if top_phrase is not None else None

        kept_phrases_ordered = (
            [phrase_by_box.get(id(k)) for k in kept]
            if raw_phrases is not None else None
        )
        subj = _compute_distinct_subjects(kept, kept_phrases_ordered)

        return DetectionResult(
            bbox_xywh_normalized=(top[0], top[1], top[2], top[3]),
            confidence=top[4],
            n_raw_detections=n_raw,
            n_distinct_detections=len(subj),
            distinct_subjects=subj,
            text_label=top_phrase,
            text_label_score=top_score,
            detection_ms=elapsed_ms,
        )
```

```python
# scripts/detect_subjects/_sam3_shared.py
"""Module-level lazy-init singleton for the facebook/sam3 model + processor.

Both Sam3Detector and Sam3Segmenter import from here so the ~2 GB weights are
loaded exactly once per process, regardless of construction order.
"""
from __future__ import annotations
import torch
from transformers import Sam3Model, Sam3Processor

SAM3_MODEL_ID = "facebook/sam3"

# Maps (device_str, dtype_str) -> (processor, model)
_SHARED_INSTANCE: dict[tuple[str, str], tuple] = {}


def _get_shared_sam3(
    device: str,
    dtype: torch.dtype,
) -> tuple[Sam3Processor, Sam3Model]:
    """Return the shared (processor, model) for the given device+dtype.

    First call loads from HuggingFace Hub and caches. Subsequent calls with
    the same device+dtype return the cached pair. Calling with a different
    device when an instance exists raises ValueError — use one device per
    process (the expected deployment pattern on a single MPS or CUDA node).
    """
    key = (device, str(dtype))
    if key in _SHARED_INSTANCE:
        return _SHARED_INSTANCE[key]

    if _SHARED_INSTANCE:
        existing_key = next(iter(_SHARED_INSTANCE))
        raise ValueError(
            f"SAM 3 singleton already initialised for {existing_key}; "
            f"cannot create second instance for {key}. "
            "Use a single device per process."
        )

    processor = Sam3Processor.from_pretrained(SAM3_MODEL_ID)
    model = (
        Sam3Model.from_pretrained(SAM3_MODEL_ID)
        .to(device=device, dtype=dtype)
    )
    model.eval()
    _SHARED_INSTANCE[key] = (processor, model)
    return processor, model
```

```python
# scripts/detect_subjects/segmenters/sam3.py
"""SAM 3 bbox-prompted segmenter.

Shares the facebook/sam3 model instance with Sam3Detector via
_sam3_shared._get_shared_sam3 to avoid loading ~2 GB of weights twice.
"""
from __future__ import annotations
import time
from typing import Optional

import numpy as np
import torch
from PIL import Image

from scripts.detect_subjects.interfaces import SegmentationResult
from scripts.detect_subjects._sam3_shared import _get_shared_sam3

SAM3_MODEL_ID = "facebook/sam3"


class Sam3Segmenter:
    model_id: str = SAM3_MODEL_ID

    def __init__(
        self,
        device: str = "mps",
        dtype: torch.dtype = torch.float32,
        **kwargs,
    ) -> None:
        self.device = device
        self.dtype = dtype
        self.processor, self.model = _get_shared_sam3(device, dtype)

    @torch.no_grad()
    def segment_with_bbox(
        self,
        image_id: str,
        image: Image.Image,
        bbox_xywh_normalized: tuple[float, float, float, float],
    ) -> SegmentationResult:
        start = time.perf_counter()

        x, y, w, h = bbox_xywh_normalized
        x1 = x * image.width
        y1 = y * image.height
        x2 = (x + w) * image.width
        y2 = (y + h) * image.height
        input_boxes = [[[x1, y1, x2, y2]]]

        inputs = self.processor(
            images=image,
            input_boxes=input_boxes,
            return_tensors="pt",
        )
        # MPS does not support float64; cast all float tensors before moving to device.
        for k, v in list(inputs.items()):
            if isinstance(v, torch.Tensor) and v.dtype == torch.float64:
                inputs[k] = v.to(torch.float32)
        inputs = inputs.to(self.device)
        if "pixel_values" in inputs:
            inputs["pixel_values"] = inputs["pixel_values"].to(self.dtype)

        outputs = self.model(**inputs)
        elapsed_ms = int((time.perf_counter() - start) * 1000)

        try:
            results = self.processor.post_process_instance_segmentation(
                outputs,
                threshold=0.0,       # accept all candidates; pick best by iou_score
                mask_threshold=0.5,
                target_sizes=[(image.height, image.width)],
            )[0]
            masks_raw = results.get("masks")
            scores_raw = results.get("scores")

            if masks_raw is None or len(masks_raw) == 0:
                return SegmentationResult(
                    mask=None, iou_score=None, segmentation_ms=elapsed_ms
                )

            scores_list = scores_raw.tolist() if scores_raw is not None else [0.0]
            best_idx = int(np.argmax(scores_list))
            mask_tensor = masks_raw[best_idx]
            if mask_tensor.ndim == 3:
                mask_tensor = mask_tensor[0]  # (C, H, W) -> (H, W)
            mask = mask_tensor.cpu().numpy().astype(bool)
            iou_score = float(scores_list[best_idx])

        except Exception:
            # Fallback to InsectSAM-style pred_masks / iou_scores if the
            # post_process API differs between checkpoint versions.
            try:
                pred_masks = outputs.pred_masks
                iou_scores_t = outputs.iou_scores
                scores_list = iou_scores_t[0].cpu().tolist()
                best_idx = int(np.argmax(scores_list))
                mask_np = pred_masks[0, best_idx].cpu().float().numpy()
                mask = (mask_np > 0.5).astype(bool)
                iou_score = float(scores_list[best_idx])
            except Exception:
                return SegmentationResult(
                    mask=None, iou_score=None, segmentation_ms=elapsed_ms
                )

        return SegmentationResult(
            mask=mask,
            iou_score=iou_score,
            segmentation_ms=elapsed_ms,
        )
```

```python
# scripts/detect_subjects/prompt_builder.py
"""DB-driven insect prompt builder.

Queries the live SQLite DB for distinct taxon orders, maps each to one or more
common-name phrases, appends life-stage phrases and negative classes, deduplicates,
sorts, and returns (phrases, version_hash).

The version hash is a stable 8-character SHA-1 fingerprint of the phrase set.
It is written to parquet rows as `prompt_version` so A/B comparisons can filter
by prompt revision without reprocessing.
"""
from __future__ import annotations
import hashlib
import sqlite3
import sys
from pathlib import Path

# ── Order -> common name mapping (parent spec §181-197) ───────────────────────
ORDER_TO_COMMON_NAMES: dict[str, list[str]] = {
    "Coleoptera":    ["a beetle"],
    "Lepidoptera":   ["a butterfly", "a moth"],
    "Hymenoptera":   ["a bee", "a wasp", "an ant"],
    "Diptera":       ["a fly"],
    "Hemiptera":     ["a true bug"],
    "Orthoptera":    ["a grasshopper", "a cricket"],
    "Odonata":       ["a dragonfly", "a damselfly"],
    "Mantodea":      ["a praying mantis"],
    "Blattodea":     ["a cockroach", "a termite"],
    "Phasmatodea":   ["a stick insect"],
    "Neuroptera":    ["a lacewing"],
    "Trichoptera":   ["a caddisfly"],
    "Ephemeroptera": ["a mayfly"],
    "Plecoptera":    ["a stonefly"],
}

# Life stages — included regardless of which orders appear in the DB
_LIFE_STAGE_PHRASES: list[str] = [
    "a caterpillar",
    "a larva",
    "a nymph",
    "a pupa",
]

# Negative classes — always included so the detector has hard distractors
NEGATIVE_CLASSES: list[str] = [
    "a flower",
    "a leaf",
    "a stem",
    "a rock",
]


def build_insect_prompt(db_path: Path) -> tuple[list[str], str]:
    """Return (prompt_phrases, version_hash) driven by taxon_orders in the DB.

    Args:
        db_path: Path to the SQLite database.

    Returns:
        A 2-tuple:
        - ``prompt_phrases``: deduplicated, sorted list of prompt strings.
        - ``version_hash``: 8-character SHA-1 of the joined phrases — stable
          across calls when DB content is unchanged.
    """
    conn = sqlite3.connect(db_path)
    try:
        rows = conn.execute(
            "SELECT DISTINCT taxon_order FROM images "
            "WHERE taxon_order IS NOT NULL AND taxon_order != ''"
        ).fetchall()
    finally:
        conn.close()

    db_orders = [r[0] for r in rows]
    phrases: set[str] = set()

    for order in db_orders:
        if order in ORDER_TO_COMMON_NAMES:
            phrases.update(ORDER_TO_COMMON_NAMES[order])
        else:
            print(
                f"[prompt_builder] WARN: taxon_order {order!r} has no entry in "
                "ORDER_TO_COMMON_NAMES — skipping",
                file=sys.stderr,
            )

    phrases.update(_LIFE_STAGE_PHRASES)
    phrases.update(NEGATIVE_CLASSES)

    sorted_phrases = sorted(phrases)
    version_hash = hashlib.sha1("|".join(sorted_phrases).encode()).hexdigest()[:8]
    return sorted_phrases, version_hash
```

```python
"""One-shot: adds prompt_version='' to existing parquet rows that lack the column.
Delete this script after running.
"""
from __future__ import annotations
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq

ROOT = Path(__file__).resolve().parent.parent
PARQUET_PATH = ROOT / "data" / "cache" / "framing_detections.parquet"


def main() -> None:
    if not PARQUET_PATH.exists():
        print(f"[backfill] {PARQUET_PATH} does not exist — nothing to do")
        return
    tbl = pq.read_table(PARQUET_PATH)
    if "prompt_version" in tbl.schema.names:
        print("[backfill] prompt_version column already present — nothing to do")
        return
    n = len(tbl)
    tbl = tbl.append_column(
        "prompt_version",
        pa.array([""] * n, type=pa.string()),
    )
    pq.write_table(tbl, PARQUET_PATH, compression="snappy")
    print(f"[backfill] added prompt_version='' to {n} rows in {PARQUET_PATH}")


if __name__ == "__main__":
    main()
```

```python
"""IoU-based label transfer from DINO+InsectSAM parquet to SAM 3 parquet.

SANITY-CHECK PROTOCOL (BEFORE TRUSTING TRANSFER):
======================================================
After SAM 3 ships and its parquet is populated with labeled images, run
this 20-image validation before applying the transfer to all labels:

1. Randomly pick 20 image_ids from the existing labels.json.
2. Manually re-label those 20 images in the validator UI under SAM 3's new
   bbox — note your Column 1 (BBox) label for each.
3. Run this script with --dry-run against only those 20 (filter the parquets
   to those image_ids, or just read the printed per-image decisions).
4. Compare auto-transferred labels to your fresh manual labels:

   ≥ 90% agreement
       → Transfer logic is trustworthy. Run --apply on the full set.
         Work through transfer_review_queue.json in the validator UI.

   70-90% agreement
       → Tighten the rules: edit MID_IOU_LO to 0.8 (removing the
         0.5-0.8 auto-transfer band so only IoU ≥ 0.8 auto-transfers).
         Re-run --dry-run, verify agreement improves, then --apply.

   < 70% agreement
       → Transfer logic is unreliable for this dataset. Do NOT run
         --apply. Fall back to full re-label of all existing images
         using the validator UI's bbox-only mode.

Transfer logic summary:
  IoU ≥ 0.8                                   → auto-transfer (any label)
  0.5 ≤ IoU < 0.8, label is "correct", and
    v2 bbox area ≥ v1 bbox area                → auto-transfer
  Otherwise                                    → re-review queue

Usage:
  # Default: dry-run, print counts only
  .venv/bin/python tools/transfer_labels.py

  # Commit changes to labels.json + write transfer_review_queue.json
  .venv/bin/python tools/transfer_labels.py --apply

  # Use non-default paths
  .venv/bin/python tools/transfer_labels.py \\

      --v1-parquet data/cache/framing_detections.parquet \\
      --v2-parquet data/cache/framing_detections.parquet \\

      --apply

import argparse
import json
import sys
import time
import tempfile
import os
from pathlib import Path
from typing import Optional

import polars as pl

from scripts.detect_subjects.config import CACHE_DIR
from scripts.detect_subjects.metrics import iou_xywh_normalized

ROOT = Path(__file__).resolve().parent.parent

DEFAULT_LABELS_PATH = CACHE_DIR / "labels.json"
DEFAULT_V1_PARQUET = CACHE_DIR / "framing_detections.parquet"
DEFAULT_V2_PARQUET = CACHE_DIR / "framing_detections.parquet"
DEFAULT_QUEUE_PATH = ROOT / "tools" / "transfer_review_queue.json"

V1_VARIANT = "grounding_dino__insectsam"
V2_VARIANT = "sam3__sam3"

# Label values treated as "correct subject" for mid-IoU auto-transfer
_CORRECT_LABELS = {"bbox_correct-subject_not-clipped"}

HIGH_IOU = 0.8
MID_IOU_LO = 0.5


def _load_parquet_bbox_index(
    parquet_path: Path, variant: str
) -> dict[str, tuple[float, float, float, float]]:
    """Return {image_id: (bbox_x, bbox_y, bbox_w, bbox_h)} for one variant."""
    df = pl.read_parquet(parquet_path).filter(pl.col("variant") == variant)
    index: dict[str, tuple[float, float, float, float]] = {}
    for row in df.iter_rows(named=True):
        if row["bbox_x"] is not None:
            index[row["image_id"]] = (
                float(row["bbox_x"]),
                float(row["bbox_y"]),
                float(row["bbox_w"]),
                float(row["bbox_h"]),
            )
    return index


def _primary_flag(label_entry: dict) -> Optional[str]:
    """Return first bbox_* flag from a label entry, or None."""
    for flag in label_entry.get("flags") or []:
        if flag.startswith("bbox_"):
            return flag
    return None


def run_transfer(
    labels_path: Path = DEFAULT_LABELS_PATH,
    v1_parquet_path: Path = DEFAULT_V1_PARQUET,
    v2_parquet_path: Path = DEFAULT_V2_PARQUET,
    queue_path: Path = DEFAULT_QUEUE_PATH,
    dry_run: bool = True,
) -> dict:
    """Run the transfer logic. Returns a summary dict.

    Summary keys:
      auto_transferred  int   — count of auto-transferred labels
      re_review         int   — count of image_ids sent to re-review queue
      skipped_no_v2     int   — labeled images absent from v2 parquet
      re_review_ids     list  — image_ids needing human re-review
    """
    labels: dict = json.loads(labels_path.read_text())
    v1_index = _load_parquet_bbox_index(v1_parquet_path, V1_VARIANT)
    v2_index = _load_parquet_bbox_index(v2_parquet_path, V2_VARIANT)

    auto_transferred: list[str] = []
    re_review: list[str] = []
    skipped_no_v2: list[str] = []
    updated_labels = dict(labels)  # shallow copy; we'll mutate entries for transferred

    now_ms = int(time.time() * 1000)

    for image_id, entry in labels.items():
        v1_bbox = v1_index.get(image_id)
        v2_bbox = v2_index.get(image_id)

        if v2_bbox is None:
            skipped_no_v2.append(image_id)
            continue

        if v1_bbox is None:
            # No v1 bbox to compare against; send to re-review to be safe
            re_review.append(image_id)
            continue

        iou = iou_xywh_normalized(v1_bbox, v2_bbox)
        primary = _primary_flag(entry)

        if iou >= HIGH_IOU:
            auto_transferred.append(image_id)
            if not dry_run:
                updated_labels[image_id] = {**entry, "transferred_at": now_ms}
        elif iou >= MID_IOU_LO:
            v1_area = v1_bbox[2] * v1_bbox[3]
            v2_area = v2_bbox[2] * v2_bbox[3]
            if primary in _CORRECT_LABELS and v2_area >= v1_area:
                auto_transferred.append(image_id)
                if not dry_run:
                    updated_labels[image_id] = {**entry, "transferred_at": now_ms}
            else:
                re_review.append(image_id)
        else:
            re_review.append(image_id)

    if not dry_run:
        # Atomic write for labels.json
        fd, tmp_path = tempfile.mkstemp(
            dir=labels_path.parent, prefix=".labels_tmp_", suffix=".json"
        )
        try:
            with os.fdopen(fd, "w") as f:
                json.dump(updated_labels, f, indent=2)
            os.replace(tmp_path, labels_path)
        except Exception:
            os.unlink(tmp_path)
            raise

        # Write re-review queue
        queue_path.parent.mkdir(parents=True, exist_ok=True)
        queue_path.write_text(json.dumps(re_review, indent=2))

    return {
        "auto_transferred": len(auto_transferred),
        "re_review": len(re_review),
        "skipped_no_v2": len(skipped_no_v2),
        "re_review_ids": re_review,
    }


def main() -> None:
    parser = argparse.ArgumentParser(description=__doc__,
                                     formatter_class=argparse.RawDescriptionHelpFormatter)
    parser.add_argument("--labels", type=Path, default=DEFAULT_LABELS_PATH)
    parser.add_argument("--v1-parquet", type=Path, default=DEFAULT_V1_PARQUET,
                        dest="v1_parquet")
    parser.add_argument("--v2-parquet", type=Path, default=DEFAULT_V2_PARQUET,
                        dest="v2_parquet")
    parser.add_argument("--queue", type=Path, default=DEFAULT_QUEUE_PATH)
    parser.add_argument("--apply", action="store_true",

        v1_parquet_path=args.v1_parquet,
        v2_parquet_path=args.v2_parquet,
        queue_path=args.queue,
        dry_run=dry_run,
    )

    total = result["auto_transferred"] + result["re_review"]
    print(f"[transfer] labeled images found in both variants: {total}")
    print(f"[transfer] auto-transferred:  {result['auto_transferred']}")
    print(f"[transfer] re-review queue:   {result['re_review']}")
    print(f"[transfer] skipped (no v2):   {result['skipped_no_v2']}")

    main()
```

```python
"""Bbox stability report between two detector variants.

Compares per-image bbox IoU across two parquet variants and emits a
markdown summary + optional histogram CSV. Used to size the re-review
queue before running tools/transfer_labels.py.

Usage:
  .venv/bin/python tools/bbox_stability.py \\
      --variant-a grounding_dino__insectsam \\
      --variant-b sam3__sam3

  # Custom parquet paths (e.g. for a knob sweep sub-run):
  .venv/bin/python tools/bbox_stability.py \\
      --variant-a grounding_dino__insectsam \\
      --variant-b sam3__sam3 \\
      --parquet-a /path/to/run_a.parquet \\
      --parquet-b /path/to/run_b.parquet
"""
from __future__ import annotations
import argparse
from datetime import date
from pathlib import Path
from typing import Optional

import polars as pl

from scripts.detect_subjects.config import CACHE_DIR
from scripts.detect_subjects.metrics import iou_xywh_normalized

ROOT = Path(__file__).resolve().parent.parent
DEFAULT_PARQUET = CACHE_DIR / "framing_detections.parquet"

STABLE_THRESHOLD = 0.8
SHIFTED_THRESHOLD = 0.5


def _load_bbox_index(
    parquet_path: Path, variant: str
) -> dict[str, tuple[float, float, float, float]]:
    df = pl.read_parquet(parquet_path).filter(pl.col("variant") == variant)
    index: dict[str, tuple[float, float, float, float]] = {}
    for row in df.iter_rows(named=True):
        if row["bbox_x"] is not None:
            index[row["image_id"]] = (
                float(row["bbox_x"]),
                float(row["bbox_y"]),
                float(row["bbox_w"]),
                float(row["bbox_h"]),
            )
    return index


def compute_stability(
    variant_a: str,
    variant_b: str,
    parquet_a: Path = DEFAULT_PARQUET,
    parquet_b: Path = DEFAULT_PARQUET,
    csv_out: Optional[Path] = None,
) -> dict:
    """Compute bbox IoU stability between two variants.

    Returns dict with keys:
      n_compared   — images present in both variants with valid bbox
      n_only_a     — images in A not in B
      n_only_b     — images in B not in A
      pct_stable   — % with IoU >= 0.8
      pct_shifted  — % with 0.5 <= IoU < 0.8
      pct_moved    — % with IoU < 0.5
      iou_values   — list of per-image IoU floats (same order as n_compared)
    """
    idx_a = _load_bbox_index(parquet_a, variant_a)
    idx_b = _load_bbox_index(parquet_b, variant_b)

    ids_a = set(idx_a)
    ids_b = set(idx_b)
    common = ids_a & ids_b

    iou_values: list[float] = []
    for image_id in sorted(common):
        iou = iou_xywh_normalized(idx_a[image_id], idx_b[image_id])
        iou_values.append(iou)

    n = len(iou_values)
    if n == 0:
        stable = shifted = moved = 0.0
    else:
        stable  = sum(1 for v in iou_values if v >= STABLE_THRESHOLD)  / n * 100
        shifted = sum(1 for v in iou_values if SHIFTED_THRESHOLD <= v < STABLE_THRESHOLD) / n * 100
        moved   = sum(1 for v in iou_values if v < SHIFTED_THRESHOLD)  / n * 100

    result = {
        "n_compared": n,
        "n_only_a": len(ids_a - ids_b),
        "n_only_b": len(ids_b - ids_a),
        "pct_stable": round(stable, 2),
        "pct_shifted": round(shifted, 2),
        "pct_moved": round(moved, 2),
        "iou_values": iou_values,
    }

    if csv_out is not None:
        # 20-bin histogram
        bin_edges = [i / 20 for i in range(21)]
        bins = [0] * 20
        for v in iou_values:
            idx = min(int(v * 20), 19)
            bins[idx] += 1
        lines = ["bucket,count"]
        for i in range(20):
            lo = round(bin_edges[i], 2)
```

```python
"""Knob sweep: vary one config threshold, report F1 + bbox stability per value.

For each candidate value the tool:
  1. Filters the parquet to rows where the chosen knob column meets the value
     (e.g. `confidence >= value` for a threshold knob).
  2. Computes F1 against existing labels using bootstrap CIs.
  3. Computes bbox stability (% IoU ≥ 0.8) between the current locked value
     and the candidate value's filtered set.

No model re-inference is performed — the tool works entirely from an existing
parquet. For a knob that changes which rows are predicted positive (thresholds),
this is correct. For a knob that changes the model itself (e.g. SAM3_BOX_THRESHOLD
during detection), you must regenerate the parquet before sweeping.

Supported knobs (operate as lower-bound thresholds on the named parquet column):
  confidence, bbox_area_ratio, bbox_long_edge_px, bbox_min_edge_px

Usage:
  .venv/bin/python tools/knob_sweep.py \\
      --knob confidence \\
      --values 0.10,0.20,0.30,0.40,0.50 \\
      --label bbox_correct-subject_not-clipped \\
      --variant sam3__sam3
"""

import json

import numpy as np

from scripts.detect_subjects.metrics import iou_xywh_normalized

# Bootstrap F1 CI (simple reimplementation; evaluation_utils may not exist yet)
def _bootstrap_f1_ci(
    y_true: list[int], y_pred: list[int], B: int = 2000, seed: int = 42
) -> tuple[float, float, float]:
    """Return (f1, ci_lo, ci_hi) using B bootstrap resamples."""
    rng = np.random.default_rng(seed)
    yt = np.array(y_true, dtype=np.int8)
    yp = np.array(y_pred, dtype=np.int8)
    n = len(yt)
    if n == 0:
        return 0.0, 0.0, 0.0

    def _f1(a: np.ndarray, b: np.ndarray) -> float:
        tp = int(((a == 1) & (b == 1)).sum())
        fp = int(((a == 0) & (b == 1)).sum())
        fn = int(((a == 1) & (b == 0)).sum())
        denom = 2 * tp + fp + fn
        return (2 * tp / denom) if denom > 0 else 0.0

    base_f1 = _f1(yt, yp)
    scores = []
    for _ in range(B):
        idx = rng.integers(0, n, size=n)
        scores.append(_f1(yt[idx], yp[idx]))
    scores.sort()
    ci_lo = float(scores[int(0.025 * B)])
    ci_hi = float(scores[int(0.975 * B)])
    return base_f1, ci_lo, ci_hi


def _load_labels(labels_path: Path) -> dict[str, dict]:
    return json.loads(labels_path.read_text())


def run_sweep(
    knob: str,
    values: list[float],
    parquet_path: Path = DEFAULT_PARQUET,
    labels_path: Path = DEFAULT_LABELS,
    variant: str = DEFAULT_VARIANT,
    target_label: str = "bbox_correct-subject_not-clipped",
    B: int = 2000,
) -> list[dict]:
    """Run the sweep and return one result dict per value.

    Each result dict has keys:
      value        — the knob value tested
      f1           — point estimate F1
      f1_ci_lo     — 2.5th percentile bootstrap F1
      f1_ci_hi     — 97.5th percentile bootstrap F1
      pct_stable   — % images with IoU ≥ 0.8 vs previous value (or first value)
      n_re_review  — count with IoU < 0.5 vs previous value
    """
    df = pl.read_parquet(parquet_path).filter(pl.col("variant") == variant)
    labels = _load_labels(labels_path)
    labeled_ids = set(labels.keys())

    # Filter to labeled rows only
    df = df.filter(pl.col("image_id").is_in(list(labeled_ids)))

    results: list[dict] = []
    prev_bbox_index: Optional[dict] = None

    for val in values:
        if knob in df.columns:
            mask = df[knob] >= val
            df_val = df.filter(mask)
        else:
            df_val = df  # Unknown knob: no filtering, just compare bboxes

        y_true: list[int] = []
        y_pred: list[int] = []
        bbox_index: dict[str, tuple[float, float, float, float]] = {}

        for row in df_val.iter_rows(named=True):
            iid = row["image_id"]
            label_entry = labels.get(iid, {})
            flags = set(label_entry.get("flags") or [])

            true_pos = 1 if target_label in flags else 0
            pred_pos = 1 if target_label in (row.get("suggested_labels") or []) else 0
            y_true.append(true_pos)
            y_pred.append(pred_pos)

            if row.get("bbox_x") is not None:
                bbox_index[iid] = (
                    float(row["bbox_x"]),
                    float(row["bbox_y"]),
                    float(row["bbox_w"]),
                    float(row["bbox_h"]),
                )

        f1, ci_lo, ci_hi = _bootstrap_f1_ci(y_true, y_pred, B=B)

        # Stability vs previous knob value's bbox set
        pct_stable = 100.0
        n_re_review = 0
        if prev_bbox_index is not None:
            common = set(prev_bbox_index) & set(bbox_index)
            if common:
                ious = [
                    iou_xywh_normalized(prev_bbox_index[iid], bbox_index[iid])
                    for iid in common
                ]
                pct_stable = sum(1 for v in ious if v >= 0.8) / len(ious) * 100
                n_re_review = sum(1 for v in ious if v < 0.5)

        results.append({
            "value": val,
            "f1": round(f1, 4),
            "f1_ci_lo": round(ci_lo, 4),
            "f1_ci_hi": round(ci_hi, 4),
            "pct_stable": round(pct_stable, 1),
            "n_re_review": n_re_review,
        })

        prev_bbox_index = bbox_index

    return results


def render_table(rows: list[dict], knob: str) -> str:
    """Render a markdown table from sweep results."""
    lines = [
        f"## Knob sweep: `{knob}`",
        "",
        "| value | F1 | F1 95% CI | stability vs prev | n_re_review |",
        "|-------|----|-----------|-------------------|-------------|",
    ]
    for r in rows:
        ci = f"({r['f1_ci_lo']:.2f}, {r['f1_ci_hi']:.2f})"
        lines.append(
            f"| {r['value']} | {r['f1']:.2f} | {ci} "
            f"| {r['pct_stable']:.0f}% | {r['n_re_review']} |"
        )
    return "\n".join(lines)


def main() -> None:
    parser = argparse.ArgumentParser(description=__doc__,
                                     formatter_class=argparse.RawDescriptionHelpFormatter)
    parser.add_argument("--knob", required=True,
                        help="Parquet column name to threshold on")
    parser.add_argument("--values", required=True,
                        help="Comma-separated float values to sweep")
    parser.add_argument("--label", default="bbox_correct-subject_not-clipped",
                        dest="target_label")
    parser.add_argument("--variant", default=DEFAULT_VARIANT)
    parser.add_argument("--parquet", type=Path, default=DEFAULT_PARQUET)
    parser.add_argument("--labels", type=Path, default=DEFAULT_LABELS)
    args = parser.parse_args()

        values=values,
        parquet_path=args.parquet,

        variant=args.variant,

    print(render_table(rows, args.knob))

    main()
```

```python
For each parquet row with text_label populated, compares the detected phrase
against the expected phrase set for that image's taxon_order (via

from __future__ import annotations

from typing import Optional

import polars as pl

from scripts.detect_subjects.config import CACHE_DIR, DATA_DIR

def _load_db_index(db_path: Path) -> dict[str, dict]:
    """Return {image_id: {"taxon_order": order}} from the DB."""
    idx: dict[str, dict] = {}
    if not db_path.exists():
        return idx
    con = sqlite3.connect(db_path)
    try:
        cur = con.execute("SELECT image_id, taxon_order FROM images")
        for image_id, taxon_order in cur:
            idx[image_id] = {"taxon_order": taxon_order or ""}
    finally:
        con.close()
    return idx


def _load_order_lookup() -> dict[str, list[str]]:
    """Import ORDER_TO_COMMON_NAMES from prompt_builder. Graceful if absent."""
    try:

        return dict(ORDER_TO_COMMON_NAMES)
    except ImportError:
        return {}


def compute_agreement(
    parquet_path: Path = DEFAULT_PARQUET,
    variant: str = DEFAULT_VARIANT,
    db_index: Optional[dict] = None,
    order_lookup: Optional[dict] = None,
    top_k: int = DEFAULT_TOP_K,
    db_path: Path = DEFAULT_DB,
) -> dict:
    """Compute phrase-vs-taxon agreement statistics.

    Returns dict with keys:
      n_evaluated       — rows with text_label present
      overall_pct       — % where text_label in expected phrase set
      per_order         — {order: {n, n_match, agreement_pct}}
      top_mismatches    — list of {image_id, actual, expected} (up to top_k)
      unmatched_orders  — orders found in DB but absent from lookup
    """
    if db_index is None:
        db_index = _load_db_index(db_path)
    if order_lookup is None:
        order_lookup = _load_order_lookup()

        pl.read_parquet(parquet_path)
        .filter(pl.col("variant") == variant)
        .filter(pl.col("text_label").is_not_null())
    )

    per_order: dict[str, dict] = {}
    mismatches: list[dict] = []
    unmatched_orders: set[str] = set()
    n_match = 0
    n_total = 0

    for row in df.iter_rows(named=True):
        image_id = row["image_id"]
        actual = row["text_label"]
        if not actual:
            continue

        db_row = db_index.get(image_id, {})
        order = db_row.get("taxon_order", "")

        expected_phrases = order_lookup.get(order)
        if expected_phrases is None:
            if order:
                unmatched_orders.add(order)
            # No expected set → skip from agreement computation
            continue

        n_total += 1
        match = actual in expected_phrases

        if order not in per_order:
            per_order[order] = {"n": 0, "n_match": 0, "agreement_pct": 0.0}
        per_order[order]["n"] += 1
        if match:
            per_order[order]["n_match"] += 1
            n_match += 1
        else:
            if len(mismatches) < top_k:
                mismatches.append({
                    "image_id": image_id,
                    "actual": actual,
                    "expected": expected_phrases,
                    "order": order,
                })

    for order, stats in per_order.items():
        stats["agreement_pct"] = (
            round(stats["n_match"] / stats["n"] * 100, 1) if stats["n"] > 0 else 0.0
        )

    overall_pct = round(n_match / n_total * 100, 1) if n_total > 0 else 0.0

    return {
        "n_evaluated": n_total,
        "overall_pct": overall_pct,
        "per_order": per_order,
        "top_mismatches": mismatches,
        "unmatched_orders": sorted(unmatched_orders),
    }


def _render_markdown(result: dict, variant: str) -> str:
    lines = [
        f"## Phrase-vs-taxon agreement: `{variant}`",
        "",
        f"Evaluated: {result['n_evaluated']} rows with text_label  "
        f"| Overall agreement: **{result['overall_pct']:.1f}%**",
        "",
        "### Agreement per order",
        "",
        "| taxon_order | n | agreement |",
        "|-------------|---|-----------|",
    ]
    for order, stats in sorted(result["per_order"].items(),
                                key=lambda x: x[1]["n"], reverse=True):
        lines.append(
            f"| {order} | {stats['n']} | {stats['agreement_pct']:.1f}% |"
        )

    if result["top_mismatches"]:
        lines += [
            "",
            "### Top mismatches",
            "",
            "| image_id | actual | expected (sample) |",
            "|----------|--------|-------------------|",
        ]
        for m in result["top_mismatches"]:
            expected_sample = ", ".join(m["expected"][:3])
            lines.append(f"| {m['image_id']} | {m['actual']} | {expected_sample} |")

    if result["unmatched_orders"]:
        lines += [
            "",
            "### Orders in DB but missing from prompt lookup",
            "",
            "These orders fall back to generic 'an insect' anchor only.",
            "",
        ]
        for order in result["unmatched_orders"]:
            lines.append(f"- {order}")

    return "\n".join(lines)


def main() -> None:
    parser = argparse.ArgumentParser(description=__doc__,
                                     formatter_class=argparse.RawDescriptionHelpFormatter)
    parser.add_argument("--variant", default=DEFAULT_VARIANT)
    parser.add_argument("--parquet", type=Path, default=DEFAULT_PARQUET)
    parser.add_argument("--db", type=Path, default=DEFAULT_DB)
    parser.add_argument("--top-k", type=int, default=DEFAULT_TOP_K)
    args = parser.parse_args()

    result = compute_agreement(
        parquet_path=args.parquet,
        variant=args.variant,
        db_path=args.db,
        top_k=args.top_k,
    )
    print(_render_markdown(result, args.variant))


if __name__ == "__main__":
    main()
```

```python
"""Ablation test: does SAM 3's mask earn its inference cost?

For each mask-dependent label (mask_blur_unusable, mask_blur_usable,
mask_poor-contrast):

  1. Train LogisticRegression(class_weight='balanced') with 5-fold CV:
       WITHOUT mask: [bbox_area_ratio, offcenter, bbox_long_edge_px,
                      subject_sharpness]
       WITH mask:    above + [mask_area_ratio, lab_delta_e,
                              boundary_sharpness, mask_iou_score]

  2. Compute ROC-AUC per fold, average.

  3. Decision rule (parent spec §343-347):
       ΔAUC = AUC_with_mask - AUC_without_mask
       ΔAUC ≥ 0.05 → keep_segmenter: True
       else         → keep_segmenter: False

Outputs:
  docs/ablation_sam3_mask.md   — human-readable report
  docs/ablation_sam3_mask.json — machine-readable for Phase 3 branch decision

Usage:
  .venv/bin/python tools/ablation_mask_features.py --variant sam3__sam3
"""
from __future__ import annotations
import argparse
import json
from pathlib import Path
from typing import Optional

import numpy as np
import polars as pl
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import StratifiedKFold
from sklearn.metrics import roc_auc_score

from scripts.detect_subjects.config import CACHE_DIR

ROOT = Path(__file__).resolve().parent.parent
DEFAULT_PARQUET = CACHE_DIR / "framing_detections.parquet"
DEFAULT_LABELS  = CACHE_DIR / "labels.json"
DEFAULT_VARIANT = "sam3__sam3"
DEFAULT_JSON_OUT = ROOT / "docs" / "ablation_sam3_mask.json"
DEFAULT_MD_OUT   = ROOT / "docs" / "ablation_sam3_mask.md"

DEFAULT_LABELS_TO_TEST = [
    "mask_blur_unusable",
    "mask_blur_usable",
    "mask_poor-contrast",
]

BBOX_FEATURES   = ["bbox_area_ratio", "offcenter", "bbox_long_edge_px",
                    "subject_sharpness"]
MASK_FEATURES   = ["mask_area_ratio", "lab_delta_e", "boundary_sharpness",
                    "mask_iou_score"]
ALL_FEATURES    = BBOX_FEATURES + MASK_FEATURES

KEEP_THRESHOLD = 0.05  # parent spec §344


def _load_labels(labels_path: Path) -> dict[str, dict]:
    return json.loads(labels_path.read_text())


def _extract_Xy(
    df: pl.DataFrame,
    labels: dict[str, dict],
    target_label: str,
    feature_cols: list[str],
) -> tuple[np.ndarray, np.ndarray]:
    """Return (X, y) arrays for labeled rows with no NaN in feature_cols."""
    X_rows: list[list[float]] = []
    y_rows: list[int] = []

    for row in df.iter_rows(named=True):
        iid = row["image_id"]
        if iid not in labels:
            continue
        flags = set(labels[iid].get("flags") or [])
        y = 1 if target_label in flags else 0

        feat = [row.get(col) for col in feature_cols]
        if any(v is None for v in feat):
            continue
        X_rows.append([float(v) for v in feat])
        y_rows.append(y)

    return np.array(X_rows, dtype=np.float64), np.array(y_rows, dtype=np.int8)


def _cv_auc(X: np.ndarray, y: np.ndarray, n_splits: int = 5) -> float:
    """5-fold stratified CV mean ROC-AUC. Returns 0.5 if fewer than 2 classes."""
    if len(set(y)) < 2:
        return 0.5
    skf = StratifiedKFold(n_splits=n_splits, shuffle=True, random_state=42)
    aucs: list[float] = []
    for train_idx, test_idx in skf.split(X, y):
        clf = LogisticRegression(class_weight="balanced", max_iter=500,
                                 random_state=42)
        clf.fit(X[train_idx], y[train_idx])
        prob = clf.predict_proba(X[test_idx])[:, 1]
        try:
            aucs.append(roc_auc_score(y[test_idx], prob))
        except ValueError:
            aucs.append(0.5)
    return float(np.mean(aucs))


def _apply_decision_rule(auc_without: float, auc_with: float) -> dict:
    delta = auc_with - auc_without
    return {
        "delta_auc": round(delta, 4),
        "keep_segmenter": delta >= KEEP_THRESHOLD,
    }


def run_ablation(
    parquet_path: Path = DEFAULT_PARQUET,
    labels_path: Path = DEFAULT_LABELS,
    variant: str = DEFAULT_VARIANT,
    labels_to_test: list[str] = DEFAULT_LABELS_TO_TEST,
    json_out: Optional[Path] = None,
    md_out: Optional[Path] = None,
    B: int = 2000,  # reserved for future bootstrap CI; CV used currently
) -> dict:
    """Run the ablation and return per-label results dict.

    Each label entry contains:
      auc_without_mask  — CV AUC with bbox features only
      auc_with_mask     — CV AUC with bbox + mask features
      delta_auc         — auc_with_mask - auc_without_mask
      keep_segmenter    — True if delta_auc >= 0.05
      n_positive        — count of positive labels
      n_total           — count of labeled rows with complete features
    """
    df = pl.read_parquet(parquet_path).filter(pl.col("variant") == variant)
    labels = _load_labels(labels_path)

    results: dict[str, dict] = {}

    for target in labels_to_test:
        X_bbox, y = _extract_Xy(df, labels, target, BBOX_FEATURES)
        X_all, y_all = _extract_Xy(df, labels, target, ALL_FEATURES)

        if len(y) < 5:
            results[target] = {
                "auc_without_mask": None,
                "auc_with_mask": None,
                "delta_auc": None,
                "keep_segmenter": None,
                "n_positive": int(y.sum()) if len(y) > 0 else 0,
                "n_total": len(y),
                "error": f"insufficient labeled rows: {len(y)} (need ≥ 5)",
            }
            continue

        auc_without = _cv_auc(X_bbox, y)
        auc_with = _cv_auc(X_all, y_all)
        decision = _apply_decision_rule(auc_without, auc_with)

        results[target] = {
            "auc_without_mask": round(auc_without, 4),
            "auc_with_mask": round(auc_with, 4),
            **decision,
            "n_positive": int(y.sum()),
            "n_total": len(y),
        }

    if json_out is not None:
        json_out.parent.mkdir(parents=True, exist_ok=True)
        json_out.write_text(json.dumps(results, indent=2))

    if md_out is not None:
        md_out.parent.mkdir(parents=True, exist_ok=True)
        md_out.write_text(_render_markdown(results, variant))

    return results


def _render_markdown(results: dict, variant: str) -> str:
    lines = [
        f"## Ablation: SAM 3 mask feature contribution (`{variant}`)",
        "",
        "Decision rule: ΔAUC ≥ 0.05 → keep segmenter  (parent spec §343-347)",
        "",
        "| label | AUC (no mask) | AUC (with mask) | ΔAUC | decision |",
        "|-------|---------------|-----------------|------|----------|",
    ]
    for label, stats in sorted(results.items()):
        if stats.get("error"):
            lines.append(
                f"| {label} | — | — | — | SKIP: {stats['error']} |"
            )
        else:
            decision = "**keep**" if stats["keep_segmenter"] else "**drop**"
            lines.append(
                f"| {label} | {stats['auc_without_mask']:.3f} "
                f"| {stats['auc_with_mask']:.3f} "
                f"| {stats['delta_auc']:+.3f} | {decision} |"
            )
    overall_keep = all(
        v.get("keep_segmenter") is True
        for v in results.values()
        if v.get("keep_segmenter") is not None
    )
    lines += [
        "",
        f"**Overall verdict: {'keep segmenter' if overall_keep else 'drop segmenter'}**",
        "",
        f"n_positive / n_total per label:",
    ]
    for label, stats in sorted(results.items()):
        lines.append(
            f"  - `{label}`: {stats.get('n_positive', '?')}"
            f" / {stats.get('n_total', '?')}"
        )
    return "\n".join(lines) + "\n"


def main() -> None:
    parser = argparse.ArgumentParser(description=__doc__,
                                     formatter_class=argparse.RawDescriptionHelpFormatter)
    parser.add_argument("--variant", default=DEFAULT_VARIANT)
    parser.add_argument("--parquet", type=Path, default=DEFAULT_PARQUET)
    parser.add_argument("--labels", type=Path, default=DEFAULT_LABELS)
    parser.add_argument("--json-out", type=Path, default=DEFAULT_JSON_OUT)
    parser.add_argument("--md-out", type=Path, default=DEFAULT_MD_OUT)
    parser.add_argument("--label", action="append", dest="labels_to_test",
                        metavar="LABEL",
                        help="Repeat to test multiple labels. "
                             f"Default: {DEFAULT_LABELS_TO_TEST}")
    args = parser.parse_args()

    labels_to_test = args.labels_to_test or DEFAULT_LABELS_TO_TEST

    results = run_ablation(
        parquet_path=args.parquet,
        labels_path=args.labels,
        variant=args.variant,
        labels_to_test=labels_to_test,
        json_out=args.json_out,
        md_out=args.md_out,
    )

    print(_render_markdown(results, args.variant))
    if args.json_out:
        print(f"\nJSON written to {args.json_out}")
    if args.md_out:
        print(f"Markdown written to {args.md_out}")


if __name__ == "__main__":
    main()
```

```python
"""Render bbox overlays for Claude's visual smoke test of detector output.

Two-tier protocol (per Phase 2 spec workflow):
  - Iterative 30-image samples DURING knob tuning loop (--n 30)
  - Final 50-image confidence check BEFORE handing to human review (--n 50)

For each sampled image: load the original from data/images/, draw the primary
bbox (pink), draw any secondaries (cyan), apply a red border if the matched
text_label is a NEGATIVE class (flower/leaf/stem/rock). Save JPEGs to the
output dir and write index.json so Claude can locate them by image_id.

Usage:
    .venv/bin/python -m tools.render_bbox_overlays \\
        --variant sam3__sam3 --n 30 --out /tmp/bbox_check_30/
"""
from __future__ import annotations
import argparse
import json
import sqlite3
import sys
from pathlib import Path
from typing import Callable, Optional

import polars as pl
from PIL import Image, ImageDraw

from scripts.detect_subjects.config import DATA_DIR, PARQUET_PATH

DB_PATH = DATA_DIR / "db" / "line-of-bugs.db"

PRIMARY_COLOR = (255, 110, 199)         # pink
SECONDARY_COLOR = (103, 212, 230)       # cyan
NEGATIVE_BORDER_COLOR = (255, 70, 70)   # red
TEXT_COLOR = (255, 255, 255)            # white

NEGATIVE_CLASS_KEYWORDS = {"flower", "leaf", "stem", "rock"}


def _resolve_image_path_via_db(image_id: str) -> Optional[Path]:
    """Look up image_id → filename via the SQLite DB."""
    if not DB_PATH.exists():
        return None
    con = sqlite3.connect(DB_PATH)
    try:
        row = con.execute(
            "SELECT filename FROM images WHERE image_id = ?", (image_id,),
        ).fetchone()
        if not row or not row[0]:
            return None
        return DATA_DIR / row[0]
    finally:
        con.close()


def sample_rows(df: pl.DataFrame, n: int, seed: int) -> pl.DataFrame:
    """Sample n rows deterministically; if df has < n, return all."""
    if df.height <= n:
        return df
    return df.sample(n=n, seed=seed, shuffle=True)


def _is_negative_class(text_label: Optional[str]) -> bool:
    if not text_label:
        return False
    stripped = text_label.strip().lower()
    for prefix in ("a ", "an ", "the "):
        if stripped.startswith(prefix):
            stripped = stripped[len(prefix):]
            break
    return any(kw == stripped or kw in stripped.split() for kw in NEGATIVE_CLASS_KEYWORDS)


def build_overlay_jpeg(
    image: Image.Image,
    bbox_xywh_normalized: Optional[tuple[float, float, float, float]],
    distinct_subjects: list,
    text_label: Optional[str] = None,
    text_label_score: Optional[float] = None,
) -> Image.Image:
    """Draw bboxes + text overlay onto a copy of the image. Returns the copy."""
    out = image.copy().convert("RGB")
    draw = ImageDraw.Draw(out)
    W, H = out.size

    # Draw secondaries first (cyan, behind primary)
    for s in distinct_subjects or []:
        if len(s) < 4:
            continue
        sx, sy, sw, sh = s[0], s[1], s[2], s[3]
        draw.rectangle(
            [sx * W, sy * H, (sx + sw) * W, (sy + sh) * H],
            outline=SECONDARY_COLOR, width=2,
        )

    # Draw primary
    if bbox_xywh_normalized is not None:
        x, y, w, h = bbox_xywh_normalized
        x1, y1, x2, y2 = x * W, y * H, (x + w) * W, (y + h) * H
        is_neg = _is_negative_class(text_label)
        color = NEGATIVE_BORDER_COLOR if is_neg else PRIMARY_COLOR
        draw.rectangle([x1, y1, x2, y2], outline=color, width=4)
        if text_label is not None:
            score_str = f"{text_label_score:.2f}" if text_label_score is not None else "?"
            label = f"{text_label}·{score_str}"
            draw.text((x1 + 4, max(0, y1 - 14)), label, fill=color)
    return out


def render_and_index(
    df: pl.DataFrame,
    n: int,
    seed: int,
    out_dir: Path,
    resolve_image_path: Callable[[str], Optional[Path]] = _resolve_image_path_via_db,
) -> list[dict]:
    """Sample n rows, render overlays, save JPEGs + index.json. Return the index list."""
    out_dir.mkdir(parents=True, exist_ok=True)
    sampled = sample_rows(df, n=n, seed=seed)

    index: list[dict] = []
    for i, row in enumerate(sampled.iter_rows(named=True)):
        image_id = row["image_id"]
        img_path = resolve_image_path(image_id)
        if img_path is None or not img_path.exists():
            index.append({
                "image_id": image_id, "overlay_jpeg": None,
                "error": f"image file not found at {img_path}",
            })
            continue

        with Image.open(img_path) as src:
            src = src.convert("RGB")
            bbox = None
            if row.get("bbox_x") is not None:
                bbox = (row["bbox_x"], row["bbox_y"], row["bbox_w"], row["bbox_h"])
            distinct = row.get("distinct_subjects") or []
            out = build_overlay_jpeg(
                src, bbox, distinct,
                text_label=row.get("text_label"),
                text_label_score=row.get("text_label_score"),
            )
            fname = f"{i:03d}_{image_id}.jpg"
            out.save(out_dir / fname, "JPEG", quality=85)
            index.append({
                "image_id": image_id,
                "overlay_jpeg": fname,
                "variant": row.get("variant"),
                "text_label": row.get("text_label"),
                "text_label_score": row.get("text_label_score"),
                "bbox_xywh": bbox,
                "n_distinct": len(distinct),
            })

    (out_dir / "index.json").write_text(json.dumps(index, indent=2))
    return index


def main() -> int:
    parser = argparse.ArgumentParser(description=__doc__.split("\n\n")[0])
    parser.add_argument("--variant", required=True, help="parquet variant tag (e.g., sam3__sam3)")
    parser.add_argument("--n", type=int, default=30, help="sample size (30 iterative / 50 final)")
    parser.add_argument("--out", type=Path, required=True, help="output directory")
    parser.add_argument("--seed", type=int, default=42, help="sampling seed")
    parser.add_argument("--seed-fresh", action="store_true",
                        help="use a fresh time-based seed (for final 50-check that shouldn't reuse 30-check images)")
    parser.add_argument("--parquet", type=Path, default=PARQUET_PATH)
    args = parser.parse_args()

    seed = args.seed
    if args.seed_fresh:
        import time
        seed = int(time.time())

    df = pl.read_parquet(args.parquet).filter(pl.col("variant") == args.variant)
    if df.height == 0:
        print(f"ERROR: no rows in parquet for variant={args.variant!r}", file=sys.stderr)
        return 1

    index = render_and_index(df, n=args.n, seed=seed, out_dir=args.out)
    n_ok = sum(1 for e in index if e.get("overlay_jpeg"))
    n_err = sum(1 for e in index if e.get("error"))
    print(f"rendered: {n_ok} ok, {n_err} errors, sample seed={seed}")
    print(f"index: {args.out / 'index.json'}")
    print(f"\nClaude: use the Read tool on each {args.out}/*.jpg to visually inspect.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
