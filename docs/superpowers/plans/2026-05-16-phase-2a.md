> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate `labels.json` to the new 4-column label vocabulary (per parent spec §219-298 of `docs/superpowers/specs/2026-05-15-modular-pipeline-design.md`), rewrite the validator UI as a 4-column layout in-place in the single HTML template, wire `gate.py` into the live pipeline (now possible because rule_labeler emits new vocab), and ship a "bbox-only labeling mode" + "stale label warning" + per-bbox text-label overlay in the UI.

**Architecture:** Phase 2a is independent of model changes — ships with the existing `grounding_dino` + `insectsam` stack. UI changes are in-place edits to ONE 583-line template file. `label_server.py` server pattern (`http.server` + atomic-write to single `labels.json`) is preserved. The biggest change is the rule labeler's output vocabulary: `rule_labeler.py:suggest_labels` and `classify_framing` will emit new label strings, and `gate.py` will consume them and write `gate_decision` to parquet.

**Tech Stack:** Python 3.12, vanilla HTML/CSS/JS (no framework), Python `http.server`, polars 1.40, pyarrow 24.0. No new dependencies.

```python
"""One-shot labels.json vocabulary migration: old vocab -> 4-column new vocab.

Usage:
  .venv/bin/python tools/migrate_labels.py           # dry run (default)
  .venv/bin/python tools/migrate_labels.py --apply   # apply changes

Backup is always written to tools/manual-labels-backups/labels-pre-phase2.json
before any mutation. The script exits non-zero if backup already exists on --apply
(prevents double-apply).

Delete this file after --apply succeeds (per repo convention for one-shot scripts).
"""
from __future__ import annotations
import argparse
import json
import os
import shutil
import sys
import tempfile
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
LABELS_PATH = ROOT / "data" / "cache" / "labels.json"
BACKUP_PATH = ROOT / "tools" / "manual-labels-backups" / "labels-pre-phase2.json"

# Keys map old flag string -> new flag string.
# None means DELETE the flag entirely (no equivalent in new vocab).
FLAG_RENAME: dict[str, str | None] = {
    "crop-too-tight":                       None,
    "crop-too-loose":                       None,
    "cropped-good":                         None,
    "original-good":                        None,
    "subject-blurred_partially-usable":     "mask_blur_usable",
    "subject-blurred_unusable":             "mask_blur_unusable",
    "subject-blurred":                      "mask_blur_unusable",
    "subject-clipped":                      "bbox_correct-subject_clipped",
    "bbox-wrong_correct-subject":           "bbox_correct-subject_clipped",
    "bbox-wrong_better-subject-other-bbox": "bbox_wrong-subject",
    "bbox-wrong":                           "bbox_correct-subject_clipped",
    "no-bug":                               "bbox-content_no-bug",
    "bug-too-small":                        "bbox-content_subject-too-small",
    "multi-bug":                            None,
    "poor-contrast":                        "mask_poor-contrast",
    "other-bad":                            "ml_other-bad",
}

PREF_RENAME: dict[str, str | None] = {
    "original": None,
    "cropped":  None,
}


def migrate_record(image_id: str, rec: dict) -> tuple[dict, list[str]]:
    """Return (new_record, list_of_human_readable_change_strings)."""
    changes: list[str] = []
    new_flags: list[str] = []
    seen: set[str] = set()

    for old_flag in rec.get("flags") or []:
        if old_flag not in FLAG_RENAME:
            if old_flag not in seen:
                new_flags.append(old_flag)
                seen.add(old_flag)
            changes.append(f"  KEPT unknown flag: {old_flag!r}")
            continue
        new_flag = FLAG_RENAME[old_flag]
        if new_flag is None:
            changes.append(f"  DROP flag: {old_flag!r}")
        else:
            if new_flag not in seen:
                new_flags.append(new_flag)
                seen.add(new_flag)
            if old_flag != new_flag:
                changes.append(f"  RENAME flag: {old_flag!r} -> {new_flag!r}")

    pref = rec.get("preference")
    if pref is not None:
        new_pref = PREF_RENAME.get(pref, pref)
        if new_pref is None:
            changes.append(f"  DROP preference: {pref!r}")
        else:
            changes.append(f"  KEPT unknown preference: {pref!r}")

    new_rec = {
        "flags": new_flags,
        "unsure": rec.get("unsure", False),
        "reviewed_at": rec.get("reviewed_at"),
        "user_edited": rec.get("user_edited", True),
        # "preference" key deleted entirely — not part of new schema
    }
    return new_rec, changes


def run(dry_run: bool) -> None:
    if not LABELS_PATH.exists():
        print(f"ERROR: {LABELS_PATH} not found.", file=sys.stderr)
        sys.exit(1)

    raw = json.loads(LABELS_PATH.read_text())
    if not isinstance(raw, dict):
        print("ERROR: labels.json is not a dict.", file=sys.stderr)
        sys.exit(1)

    print(f"Loaded {len(raw)} label entries from {LABELS_PATH}")

    if not dry_run:
        if BACKUP_PATH.exists():
            print(
                f"ERROR: backup already exists at {BACKUP_PATH}. "
                "Double-apply guard triggered — remove backup manually if intentional.",
                file=sys.stderr,
            )
            sys.exit(1)
        BACKUP_PATH.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(LABELS_PATH, BACKUP_PATH)
        print(f"Backup written to {BACKUP_PATH}")

    new_data: dict[str, dict] = {}
    rename_count = 0
    drop_count = 0
    drop_pref_count = 0

    for image_id, rec in sorted(raw.items()):
        new_rec, changes = migrate_record(image_id, rec)
        if changes:
            print(f"{image_id}:")
            for c in changes:
                print(c)
            rename_count += sum(1 for c in changes if "RENAME" in c)
            drop_count += sum(1 for c in changes if "DROP flag" in c)
            drop_pref_count += sum(1 for c in changes if "DROP preference" in c)
        new_data[image_id] = new_rec

    print(f"\nSummary: {rename_count} renames, {drop_count} flag drops, "
          f"{drop_pref_count} preference drops across {len(raw)} entries.")

    if dry_run:
        print("\nDRY RUN -- no changes written. Re-run with --apply to apply.")
        return

    fd, tmp = tempfile.mkstemp(prefix="labels.", suffix=".json.tmp",
                                dir=str(LABELS_PATH.parent))
    try:
        with os.fdopen(fd, "w") as f:
            json.dump(new_data, f, indent=2)
        os.replace(tmp, LABELS_PATH)
    except Exception:
        try:
            os.unlink(tmp)
        except FileNotFoundError:
            pass
        raise
    print(f"\nApplied. {LABELS_PATH} updated.")
    print("Next step: delete this script (per repo convention for one-shot scripts).")


if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--apply", action="store_true",
                    help="Apply changes (default: dry run only).")
    args = ap.parse_args()
    run(dry_run=not args.apply)
```

```python
"""Compute the set of labels that describe a detection.

New label vocabulary (4-column, Phase 2a):
  Column 2 -- BBox Content (amber):
    bbox-content_no-bug           (no detection or low confidence)
    bbox-content_single           (one bug in bbox, no size/count problems)
    bbox-content_bbox-multibug_unusable   (>=2 bug centers inside primary bbox)
    bbox-content_subject-too-small        (bbox_long_edge_px < 512 or area tiny)
    bbox-content_image-multi-bug  (informational: >=2 distinct in image)
  Column 3 -- Mask Rule (sky-blue):
    mask_poor-contrast            (lab_delta_e < threshold)

Column 1 (bbox_*) is human-set only.
Column 4 (ml_*) is human-set only in Phase 2a.

Note: mask blur labels are NOT auto-suggested. Laplacian variance is unreliable
on uniform-textured bugs; users flag manually via the validator UI.
"""
from __future__ import annotations
from typing import Optional

from scripts.detect_subjects.config import (
    CLASSIFY_HIDDEN_CONF,
    CLASSIFY_HIDDEN_AREA,
    CLASSIFY_WIDE_AREA,
    CLASSIFY_CAMOUFLAGED_DELTA,
    CLASSIFY_BUG_TOO_SMALL_EDGE_PX,
)


def count_bugs_in_primary_bbox(
    primary_bbox: tuple[float, float, float, float],
    all_high_conf_detections: list[tuple],
) -> int:
    """Count detections whose center falls inside primary_bbox.

    primary_bbox: (x, y, w, h) normalized coords.
    all_high_conf_detections: list of (x, y, w, h, conf, ...) tuples.
    Boundary is inclusive on both edges.

    Per parent spec section 244-256: if count >= 2 emit bbox-content_bbox-multibug_unusable.
    """
    px, py, pw, ph = primary_bbox
    count = 0
    for det in all_high_conf_detections:
        dx, dy, dw, dh = det[0], det[1], det[2], det[3]
        cx = dx + dw / 2.0
        cy = dy + dh / 2.0
        if px <= cx <= px + pw and py <= cy <= py + ph:
            count += 1
    return count


def suggest_labels(
    confidence: Optional[float],
    bbox_area_ratio: Optional[float],
    bbox_long_edge_px: Optional[float],
    n_distinct_detections: int,
    n_in_primary_bbox: int,
    mask_area_ratio: Optional[float],
    lab_delta_e: Optional[float],
    bbox_touches_edge: Optional[bool] = None,  # unused post-2a; kept for compat
) -> list[str]:
    """Return list of new-vocab label strings for this detection.

    Returns Column 2 (bbox-content_*) and Column 3 (mask_*) labels.
    Column 1 and 4 are human-only and not returned here.
    """
    if confidence is None or bbox_area_ratio is None or confidence < CLASSIFY_HIDDEN_CONF:
        return ["bbox-content_no-bug"]

    out: list[str] = []

    # Column 2: BBox Content
    too_small = (
        bbox_area_ratio < CLASSIFY_HIDDEN_AREA
        or (bbox_long_edge_px is not None and bbox_long_edge_px < CLASSIFY_BUG_TOO_SMALL_EDGE_PX)
    )
    if too_small:
        out.append("bbox-content_subject-too-small")

    if n_in_primary_bbox >= 2:
        out.append("bbox-content_bbox-multibug_unusable")
    elif n_distinct_detections >= 2:
        # Image-level multi-detection is informational only (not a gate signal).
        out.append("bbox-content_image-multi-bug")

    has_content_rejection = any(
        lbl in out for lbl in (
            "bbox-content_no-bug",
            "bbox-content_bbox-multibug_unusable",
            "bbox-content_subject-too-small",
        )
    )
    if not has_content_rejection:
        out.append("bbox-content_single")

    # Column 3: Mask Rule
    if (mask_area_ratio is not None and lab_delta_e is not None
            and lab_delta_e < CLASSIFY_CAMOUFLAGED_DELTA):
        out.append("mask_poor-contrast")

    return out


def primary_label(labels: list[str]) -> str:
    """Pick the highest-priority label for badge / stats use."""
    priority = [
        "bbox-content_no-bug",
        "bbox-content_subject-too-small",
        "bbox-content_bbox-multibug_unusable",
        "mask_poor-contrast",
        "bbox-content_image-multi-bug",
        "bbox-content_single",
    ]
    for p in priority:
        if p in labels:
            return p
    return labels[0] if labels else "bbox-content_no-bug"


# Legacy framing_quality string -- maps new primary label to old underscore enum.
# The framing_quality parquet column stays through Phase 2 (dropped in Phase 3).
_PRIMARY_TO_FRAMING = {
    "bbox-content_no-bug":                 "no_bug",
    "bbox-content_subject-too-small":       "bug_too_small",
    "bbox-content_bbox-multibug_unusable":  "multi_bug",
    "mask_poor-contrast":                   "poor_contrast",
    "bbox-content_image-multi-bug":         "good",
    "bbox-content_single":                  "good",
}


def classify_framing(
    confidence: Optional[float],
    bbox_area_ratio: Optional[float],
    bbox_long_edge_px: Optional[float],
    n_distinct_detections: int,
    mask_area_ratio: Optional[float],
    lab_delta_e: Optional[float],
    bbox_touches_edge: Optional[bool] = None,
    n_in_primary_bbox: int = 1,
) -> str:
    """Single-label primary classification for backward compat / stats.

    Returns legacy underscore strings (no_bug, bug_too_small, multi_bug,
    poor_contrast, wide, tight, good) stored in the framing_quality column.
    """
    labels = suggest_labels(
        confidence=confidence,
        bbox_area_ratio=bbox_area_ratio,
        bbox_long_edge_px=bbox_long_edge_px,
        n_distinct_detections=n_distinct_detections,
        n_in_primary_bbox=n_in_primary_bbox,
        mask_area_ratio=mask_area_ratio,
        lab_delta_e=lab_delta_e,
        bbox_touches_edge=bbox_touches_edge,
    )
    prim = primary_label(labels)
    # Finer-grained legacy buckets for "single" / informational cases.
    if prim in ("bbox-content_single", "bbox-content_image-multi-bug"):
        if bbox_area_ratio is not None and bbox_area_ratio > 0.50:
            return "tight"
        if bbox_area_ratio is not None and bbox_area_ratio < CLASSIFY_WIDE_AREA:
            return "wide"
        return "good"
    return _PRIMARY_TO_FRAMING.get(prim, "good")
```

```python
"""Regression: compare current parquet against phase 2a baseline.

Expected deltas from baseline after 2a:
  - suggested_labels: new-vocab strings only (bbox-content_*, mask_*)
  - gate_decision: populated ("keep" or "reject")
  - framing_quality: may change (legacy mapping updated)
  - All other columns: identical

Skip if baseline doesn't exist (run T0 first).
"""
from __future__ import annotations
from pathlib import Path

import pytest
import polars as pl

BASELINE_PATH = Path("tests/python/_phase2a_baseline/baseline.parquet")
CURRENT_PATH = Path("data/cache/framing_detections.parquet")

NEW_VOCAB = {
    "bbox-content_no-bug", "bbox-content_single",
    "bbox-content_bbox-multibug_unusable", "bbox-content_bbox-multibug_usable",
    "bbox-content_subject-too-small", "bbox-content_image-multi-bug",
    "mask_poor-contrast", "mask_blur_unusable", "mask_blur_usable",
}
OLD_VOCAB = {
    "no-bug", "bug-too-small", "multi-bug", "poor-contrast",
    "subject-clipped", "cropped-good", "original-good",
}


@pytest.mark.skipif(not BASELINE_PATH.exists(), reason="baseline not captured — run T0 first")
@pytest.mark.skipif(not CURRENT_PATH.exists(), reason="parquet not present")
def test_suggested_labels_use_new_vocab():
    df = pl.read_parquet(CURRENT_PATH)
    all_labels = df["suggested_labels"].explode().drop_nulls().to_list()
    leaked = set(all_labels) & OLD_VOCAB
    assert not leaked, f"Old vocab leaked into suggested_labels: {leaked}"
    unknown = set(all_labels) - NEW_VOCAB
    assert not unknown, f"Unknown label strings (not in expected new vocab): {unknown}"


@pytest.mark.skipif(not BASELINE_PATH.exists(), reason="baseline not captured — run T0 first")
@pytest.mark.skipif(not CURRENT_PATH.exists(), reason="parquet not present")
def test_gate_decision_populated():
    df = pl.read_parquet(CURRENT_PATH)
    if "gate_decision" not in df.columns:
        pytest.skip("gate_decision column not yet present")
    null_count = df["gate_decision"].null_count()
    assert null_count == 0, f"{null_count} rows have null gate_decision"
    valid = {"keep", "reject"}
    unique = set(df["gate_decision"].unique().to_list())
    assert unique <= valid, f"Unexpected gate_decision values: {unique - valid}"


@pytest.mark.skipif(not BASELINE_PATH.exists(), reason="baseline not captured — run T0 first")
@pytest.mark.skipif(not CURRENT_PATH.exists(), reason="parquet not present")
def test_stable_columns_unchanged():
    """Columns not touched by Phase 2a must match baseline exactly."""
    baseline = pl.read_parquet(BASELINE_PATH)
    current = pl.read_parquet(CURRENT_PATH)
    changed = {"suggested_labels", "gate_decision", "framing_quality"}
    assert len(current) == len(baseline), (
        f"Row count changed: baseline={len(baseline)}, current={len(current)}"
    )
    common = set(baseline.columns) & set(current.columns) - changed
    b = baseline.sort(["image_id", "variant"])
    c = current.sort(["image_id", "variant"])
    for col in sorted(common):
        b_col = b[col].to_list()
        c_col = c[col].to_list()
        mismatches = [
            (i, bv, cv) for i, (bv, cv) in enumerate(zip(b_col, c_col)) if bv != cv
        ]
        assert not mismatches, (
            f"Column '{col}' differs in {len(mismatches)} rows. "
            f"First at index {mismatches[0][0]}: baseline={mismatches[0][1]!r}, "
            f"current={mismatches[0][2]!r}"
        )
```
