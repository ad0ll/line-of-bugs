```python
#!/usr/bin/env python3
"""One-shot migration: backfill Phase 2 columns into existing parquet rows.

Adds:
  text_label       -> None  (string, nullable)
  text_label_score -> None  (float32, nullable)
  gate_decision    -> None  (string, nullable)
  distinct_subjects -> []   (list of struct)

Run once, then delete this script per repo convention.

Usage:
    .venv/bin/python tools/migrate_parquet_schema_v2.py [--parquet PATH]
"""
from __future__ import annotations
import argparse
import shutil
import sys
import tempfile
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq

ROOT = Path(__file__).resolve().parent.parent
DEFAULT_PARQUET = ROOT / "data" / "cache" / "framing_detections.parquet"

DISTINCT_SUBJECTS_TYPE = pa.list_(pa.struct([
    ("x", pa.float32()),
    ("y", pa.float32()),
    ("w", pa.float32()),
    ("h", pa.float32()),
    ("conf", pa.float32()),
    ("phrase", pa.string()),
]))


def migrate(parquet_path: Path) -> None:
    print(f"[migrate] reading {parquet_path}")
    table = pq.read_table(parquet_path)
    n = table.num_rows
    print(f"[migrate] {n} rows, {len(table.schema)} columns")

    existing_names = set(table.schema.names)
    new_columns = []

    if "text_label" not in existing_names:
        new_columns.append(("text_label", pa.array([None] * n, type=pa.string())))
        print("[migrate] adding text_label (all None)")
    else:
        print("[migrate] text_label already present, skipping")

    if "text_label_score" not in existing_names:
        new_columns.append(("text_label_score", pa.array([None] * n, type=pa.float32())))
        print("[migrate] adding text_label_score (all None)")
    else:
        print("[migrate] text_label_score already present, skipping")

    if "gate_decision" not in existing_names:
        new_columns.append(("gate_decision", pa.array([None] * n, type=pa.string())))
        print("[migrate] adding gate_decision (all None)")
    else:
        print("[migrate] gate_decision already present, skipping")

    if "distinct_subjects" not in existing_names:
        empty_list_arr = pa.array([[] for _ in range(n)], type=DISTINCT_SUBJECTS_TYPE)
        new_columns.append(("distinct_subjects", empty_list_arr))
        print("[migrate] adding distinct_subjects (all [])")
    else:
        print("[migrate] distinct_subjects already present, skipping")

    if not new_columns:
        print("[migrate] nothing to do; parquet already has all Phase 2 columns")
        return

    result = table
    for col_name, col_array in new_columns:
        result = result.append_column(
            pa.field(col_name, col_array.type),
            col_array,
        )

    # Atomic write: write to temp file in same dir, then rename
    tmp_fd, tmp_path = tempfile.mkstemp(
        prefix="migrate_parquet_v2.", suffix=".parquet.tmp",
        dir=str(parquet_path.parent),
    )
    import os
    os.close(tmp_fd)
    try:
        pq.write_table(result, tmp_path, compression="snappy")
        shutil.move(tmp_path, str(parquet_path))
    except Exception:
        try:
            os.unlink(tmp_path)
        except FileNotFoundError:
            pass
        raise

    print(f"[migrate] wrote {result.num_rows} rows, {len(result.schema)} columns to {parquet_path}")
    print("[migrate] done. Delete this script: tools/migrate_parquet_schema_v2.py")


def main() -> None:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--parquet", default=str(DEFAULT_PARQUET),
                    help="Path to framing_detections.parquet")
    args = ap.parse_args()
    path = Path(args.parquet)
    if not path.exists():
        print(f"[migrate] ERROR: {path} does not exist", file=sys.stderr)
        sys.exit(1)
    migrate(path)


if __name__ == "__main__":
    main()
```

```python
    @torch.no_grad()
    def detect(self, image: Image.Image, image_id: str | None = None) -> DetectionResult:
        start = time.perf_counter()

        cache_key = _dino_cache_key(self.prompt, DINO_MODEL_ID)
        cache_path = DINO_CACHE_DIR / f"{image_id}__{cache_key}.json" if image_id else None
        cached = None
        if cache_path is not None and cache_path.exists():
            try:
                cached = json.loads(cache_path.read_text())
            except Exception:
                cached = None

        if cached is not None:
            boxes = cached["boxes"]
            scores = cached["scores"]
            text_scores = cached.get("text_scores", [None] * len(boxes))
            labels = cached.get("labels", [None] * len(boxes))
        else:
            inputs = self.processor(
                images=image, text=self.prompt, return_tensors="pt"
            ).to(self.device)
            if "pixel_values" in inputs:
                inputs["pixel_values"] = inputs["pixel_values"].to(self.dtype)
            outputs = self.model(**inputs)
            results = self.processor.post_process_grounded_object_detection(
                outputs,
                threshold=BOX_THRESHOLD,
                text_threshold=TEXT_THRESHOLD,
                target_sizes=[(image.height, image.width)],
            )[0]
            boxes = results["boxes"].cpu().tolist()
            scores = results["scores"].cpu().tolist()
            # `labels` is a list of matched phrase strings (one per detection).
            # `text_scores` is the per-detection text-alignment score (float).
            # Both are exposed by HF GroundingDINO post-processing when text_threshold is set.
            labels = [str(l) for l in results.get("labels", [None] * len(boxes))]
            logits = results.get("logits", None)
            if logits is not None:
                # logits shape: [num_detections, num_phrases]; take max per detection
                text_scores = logits.cpu().max(dim=-1).values.tolist()
            else:
                text_scores = [None] * len(boxes)
            if cache_path is not None:
                try:
                    cache_path.write_text(json.dumps({
                        "boxes": boxes,
                        "scores": scores,
                        "text_scores": text_scores,
                        "labels": labels,
                    }))
                except Exception:
                    pass

        elapsed_ms = int((time.perf_counter() - start) * 1000)
        return self._detect_from_raw(
            boxes=boxes, scores=scores, text_scores=text_scores,
            labels=labels, elapsed_ms=elapsed_ms,

    def _detect_from_raw(
        self,
        boxes: list,
        scores: list,
        elapsed_ms: int,
        image_w: int,
        image_h: int,
        text_scores: list | None = None,
        labels: list | None = None,
    ) -> DetectionResult:
        """Core detection logic, separated for testability."""
        if text_scores is None:
            text_scores = [None] * len(boxes)
        if labels is None:
            labels = [None] * len(boxes)

        n_raw = len(boxes)

        if not boxes:
            return DetectionResult(
                bbox_xywh_normalized=None, confidence=None,
                n_raw_detections=0, n_distinct_detections=0,
                detection_ms=elapsed_ms,
                distinct_subjects=[],
                text_label=None,
                text_label_score=None,
            )

        # Normalize pixel coords to [0,1] xywh; carry text_score + label through
        normalized = []
        for idx, ((x1, y1, x2, y2), score) in enumerate(zip(boxes, scores)):
            normalized.append((
                x1 / image_w,
                y1 / image_h,
                (x2 - x1) / image_w,
                (y2 - y1) / image_h,
                float(score),
                text_scores[idx],   # 6th: text_score (float or None)
                labels[idx],        # 7th: matched phrase string or None
            ))

        normalized.sort(key=lambda r: r[4], reverse=True)
        kept: list[tuple] = []
        for cand in normalized:
            cx, cy, cw, ch, cs = cand[0], cand[1], cand[2], cand[3], cand[4]
            if any(
                iou_xywh_normalized((cx, cy, cw, ch), (k[0], k[1], k[2], k[3]))
                > NMS_IOU_THRESHOLD
                for k in kept
            ):
                continue
            kept.append(cand)

        # Bark-beetle fix: prefer larger high-confidence bbox over top-1
        top_conf = kept[0][4]
        candidates = [
            k for k in kept
            if k[4] >= top_conf - BBOX_CONF_TOLERANCE
            and (k[2] * k[3]) <= BBOX_MAX_AREA_RATIO
        ]
        top = max(candidates, key=lambda k: k[2] * k[3]) if candidates else kept[0]

        # Distinct subject clustering
        confident = [
            k for k in kept
            if k[4] >= HIGH_CONF_THRESHOLD
            and 0.005 <= (k[2] * k[3]) <= BBOX_MAX_AREA_RATIO
        ]
        distinct_subjects: list[
            tuple[float, float, float, float, float, Optional[str]]
        ] = []
        for c in confident:
            cx_center = c[0] + c[2] / 2.0
            cy_center = c[1] + c[3] / 2.0
            inside_any = False
            for s in distinct_subjects:
                if (s[0] <= cx_center <= s[0] + s[2]
                        and s[1] <= cy_center <= s[1] + s[3]):
                    inside_any = True
                    break
            if not inside_any:
                # 6th slot: phrase from DINO's matched label (index 6 in our tuple)
                phrase = c[6] if len(c) > 6 else None
                distinct_subjects.append((c[0], c[1], c[2], c[3], c[4], phrase))

        n_distinct = len(distinct_subjects)

        # Primary bbox phrase and text alignment score
        primary_text_label = top[6] if len(top) > 6 else None
        primary_text_score = top[5] if len(top) > 5 else None
        if primary_text_score is not None:
            primary_text_score = float(primary_text_score)

        return DetectionResult(
            bbox_xywh_normalized=(top[0], top[1], top[2], top[3]),
            confidence=top[4],
            n_raw_detections=n_raw,
            n_distinct_detections=n_distinct,
            detection_ms=elapsed_ms,
            distinct_subjects=distinct_subjects,
            text_label=primary_text_label,
            text_label_score=primary_text_score,
        )
```

```python
#!/usr/bin/env python3
"""One-shot migration: rename v1_dino_insectsam → grounding_dino__insectsam in parquet.

The new name matches cfg.variant_tag() output format. After this runs,
V1_NAME in classify.py is updated to the new string so the backward-compat
constant stays accurate.

Run once against the live parquet, then delete this script.

Usage:
    .venv/bin/python tools/retag_legacy_variants.py [--parquet PATH] [--dry-run]
"""
from __future__ import annotations
import argparse
import os
import shutil
import sys
import tempfile
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq
import pyarrow.compute as pc

ROOT = Path(__file__).resolve().parent.parent
DEFAULT_PARQUET = ROOT / "data" / "cache" / "framing_detections.parquet"

OLD_TAG = "v1_dino_insectsam"
NEW_TAG = "grounding_dino__insectsam"


def retag(parquet_path: Path, dry_run: bool = False) -> None:
    print(f"[retag] reading {parquet_path}")
    table = pq.read_table(parquet_path)
    variant_col = table.column("variant")
    old_count = pc.sum(pc.equal(variant_col, OLD_TAG)).as_py()
    new_count = pc.sum(pc.equal(variant_col, NEW_TAG)).as_py()
    print(f"[retag] rows with old tag '{OLD_TAG}': {old_count}")
    print(f"[retag] rows with new tag '{NEW_TAG}': {new_count}")

    if old_count == 0:
        print("[retag] nothing to do; all rows already use new tag")
        return

    if dry_run:
        print(f"[retag] DRY RUN: would rename {old_count} rows → '{NEW_TAG}'")
        return

    # Replace OLD_TAG with NEW_TAG in the variant column
    new_variant = pc.if_else(
        pc.equal(variant_col, OLD_TAG),
        pa.scalar(NEW_TAG, type=pa.string()),
        variant_col,
    )
    col_idx = table.schema.get_field_index("variant")
    result = table.set_column(col_idx, pa.field("variant", pa.string()), new_variant)

    # Atomic write
    tmp_fd, tmp_path = tempfile.mkstemp(
        prefix="retag_variants.", suffix=".parquet.tmp",
        dir=str(parquet_path.parent),
    )
    os.close(tmp_fd)
    try:
        pq.write_table(result, tmp_path, compression="snappy")
        shutil.move(tmp_path, str(parquet_path))
    except Exception:
        try:
            os.unlink(tmp_path)
        except FileNotFoundError:
            pass
        raise

    print(f"[retag] renamed {old_count} rows to '{NEW_TAG}' in {parquet_path}")
    print(f"[retag] done. Update V1_NAME in classify.py, then delete: tools/retag_legacy_variants.py")


def main() -> None:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--parquet", default=str(DEFAULT_PARQUET))
    ap.add_argument("--dry-run", action="store_true",
                    help="Print what would happen without modifying anything")
    args = ap.parse_args()
    path = Path(args.parquet)
    if not path.exists():
        print(f"[retag] ERROR: {path} does not exist", file=sys.stderr)
        sys.exit(1)
    retag(path, dry_run=args.dry_run)


if __name__ == "__main__":
    main()
```

```python
"""Shared evaluation utilities for Phase 2+ pipeline assessment.

Pure functions, no I/O. Both bootstrap_f1_ci and pr_curve_per_label are
used by evaluate_pipeline.py (Phase 2a) and the ablation tool (Phase 2b).

Functions:
    bootstrap_f1_ci(y_true, y_pred, B, alpha) -> (f1, ci_low, ci_high)
    pr_curve_per_label(y_true, y_score, thresholds) -> list[dict]
"""
from __future__ import annotations
import random


def _f1_score(y_true: list[int], y_pred: list[int]) -> float:
    """Compute binary F1. Returns 0.0 when both precision and recall are 0."""
    tp = sum(1 for t, p in zip(y_true, y_pred) if t == 1 and p == 1)
    fp = sum(1 for t, p in zip(y_true, y_pred) if t == 0 and p == 1)
    fn = sum(1 for t, p in zip(y_true, y_pred) if t == 1 and p == 0)
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    if precision + recall == 0:
        return 0.0
    return 2 * precision * recall / (precision + recall)


def bootstrap_f1_ci(
    y_true: list[int],
    y_pred: list[int],
    B: int = 2000,
    alpha: float = 0.05,
    seed: int = 42,
) -> tuple[float, float, float]:
    """Compute binary F1 with bootstrap confidence interval.

    Args:
        y_true: ground-truth binary labels (0 or 1)
        y_pred: predicted binary labels (0 or 1)
        B: number of bootstrap resamples
        alpha: significance level (0.05 → 95% CI)
        seed: random seed for reproducibility

    Returns:
        (f1, ci_low, ci_high): point estimate + (1-alpha) confidence interval
    """
    n = len(y_true)
    f1 = _f1_score(y_true, y_pred)

    rng = random.Random(seed)
    indices = list(range(n))
    boot_f1s: list[float] = []
    for _ in range(B):
        sample_idx = [rng.choice(indices) for _ in range(n)]
        bt = [y_true[i] for i in sample_idx]
        bp = [y_pred[i] for i in sample_idx]
        boot_f1s.append(_f1_score(bt, bp))

    boot_f1s.sort()
    lo_idx = int((alpha / 2) * B)
    hi_idx = int((1 - alpha / 2) * B) - 1
    lo_idx = max(0, lo_idx)
    hi_idx = min(B - 1, hi_idx)
    return f1, boot_f1s[lo_idx], boot_f1s[hi_idx]


def pr_curve_per_label(
    y_true: list[int],
    y_score: list[float],
    thresholds: list[float],
) -> list[dict]:
    """Compute precision, recall, and F1 at each threshold.

    Args:
        y_true: ground-truth binary labels (0 or 1)
        y_score: predicted scores (higher = more likely positive)
        thresholds: list of score thresholds to evaluate

    Returns:
        List of dicts, one per threshold, each with keys:
        {threshold: float, precision: float, recall: float, f1: float}
    """
    results = []
    for t in thresholds:
        y_pred = [1 if s >= t else 0 for s in y_score]
        tp = sum(1 for yt, yp in zip(y_true, y_pred) if yt == 1 and yp == 1)
        fp = sum(1 for yt, yp in zip(y_true, y_pred) if yt == 0 and yp == 1)
        fn = sum(1 for yt, yp in zip(y_true, y_pred) if yt == 1 and yp == 0)
        precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
        if precision + recall > 0:
            f1 = 2 * precision * recall / (precision + recall)
        else:
            f1 = 0.0
        results.append({
            "threshold": t,
            "precision": precision,
            "recall": recall,
```

```python
#!/usr/bin/env python3
"""Migrate labels.json to the Phase 2a label vocabulary.

IMPORTANT: The rename mapping table is NOT yet defined here — that is a Phase 2a
task (requires finalizing the 4-column vocab with ad0ll). This script contains
the full migration infrastructure (atomic write, backup, dry-run mode) so Phase
2a only needs to fill in RENAME_MAP and DROP_LABELS.

Current state: --dry-run works (reports what WOULD be done). --apply raises
NotImplementedError because the mapping table is empty.

Usage:
    .venv/bin/python tools/migrate_labels.py --dry-run [--labels PATH]
    .venv/bin/python tools/migrate_labels.py --apply   [--labels PATH]   # Phase 2a only

Delete after running in Phase 2a per repo convention.
"""
from __future__ import annotations
import argparse
import json
import os
import shutil
import sys
import tempfile
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
DEFAULT_LABELS = ROOT / "data" / "cache" / "labels.json"
BACKUP_DIR = ROOT / "tools" / "manual-labels-backups"

# ─── Phase 2a: fill in RENAME_MAP and DROP_LABELS ──────────────────
# RENAME_MAP: {old_label_string: new_label_string}
# DROP_LABELS: set of label strings to drop entirely (not renamed)
# No rename mapping defined yet — fill in for Phase 2a.
RENAME_MAP: dict[str, str] = {}
DROP_LABELS: set[str] = set()
# ───────────────────────────────────────────────────────────────────


def _collect_all_labels(labels: dict) -> set[str]:
    """Return the set of all label strings present across all image records."""
    all_labels: set[str] = set()
    for rec in labels.values():
        if isinstance(rec, dict):
            for flag in rec.get("flags") or []:
                all_labels.add(flag)
            if rec.get("preference"):
                all_labels.add(rec["preference"])
    return all_labels


def _migrate_record(rec: dict) -> dict:
    """Apply RENAME_MAP and DROP_LABELS to a single image label record."""
    new_rec = dict(rec)
    # Migrate flags
    old_flags = list(rec.get("flags") or [])
    new_flags = []
    for f in old_flags:
        if f in DROP_LABELS:
            continue
        new_flags.append(RENAME_MAP.get(f, f))
    new_rec["flags"] = new_flags
    # Migrate preference
    pref = rec.get("preference")
    if pref is not None:
        if pref in DROP_LABELS:
            new_rec["preference"] = None
        else:
            new_rec["preference"] = RENAME_MAP.get(pref, pref)
    return new_rec


def run(labels_path: Path, dry_run: bool) -> None:
    if not labels_path.exists():
        print(f"[migrate_labels] ERROR: {labels_path} not found", file=sys.stderr)
        sys.exit(1)

    labels: dict = json.loads(labels_path.read_text() or "{}")
    print(f"[migrate_labels] loaded {len(labels)} image records from {labels_path}")

    if not RENAME_MAP and not DROP_LABELS:
        print("[migrate_labels] WARNING: RENAME_MAP and DROP_LABELS are both empty.")
        print("[migrate_labels] This is expected for Phase 2-prep skeleton.")
        print("[migrate_labels] Fill in the mapping table before running --apply.")

    all_labels = _collect_all_labels(labels)
    rename_count = sum(1 for l in all_labels if l in RENAME_MAP)
    drop_count = sum(1 for l in all_labels if l in DROP_LABELS)
    unmatched = all_labels - set(RENAME_MAP) - DROP_LABELS - set(RENAME_MAP.values())

    print(f"[migrate_labels] labels in file: {sorted(all_labels)}")
    print(f"[migrate_labels] would rename {rename_count} label type(s): {dict((k, RENAME_MAP[k]) for k in RENAME_MAP if k in all_labels)}")
    print(f"[migrate_labels] would drop {drop_count} label type(s): {DROP_LABELS & all_labels}")
    print(f"[migrate_labels] passthrough (no change): {unmatched}")

    if dry_run:
        print("[migrate_labels] DRY RUN complete — no files written")
        return

    # --apply path: raise NotImplementedError until Phase 2a fills in the mapping
    if not RENAME_MAP and not DROP_LABELS:
        raise NotImplementedError(
            "RENAME_MAP and DROP_LABELS are not yet defined. "
            "This is a Phase 2-prep skeleton. Fill in tools/migrate_labels.py "
            "RENAME_MAP and DROP_LABELS before using --apply."
        )

    # Backup before any write
    BACKUP_DIR.mkdir(parents=True, exist_ok=True)
    backup_path = BACKUP_DIR / "labels-pre-phase2.json"
    shutil.copy2(str(labels_path), str(backup_path))
    print(f"[migrate_labels] backed up to {backup_path}")

    # Apply migration
    migrated = {image_id: _migrate_record(rec) for image_id, rec in labels.items()}

    # Atomic write
    tmp_fd, tmp_path = tempfile.mkstemp(
        prefix="migrate_labels.", suffix=".json.tmp",
        dir=str(labels_path.parent),
    )
    os.close(tmp_fd)
    try:
        with open(tmp_path, "w") as f:
            json.dump(migrated, f, indent=2)
        os.replace(tmp_path, str(labels_path))
    except Exception:
        try:
            os.unlink(tmp_path)
        except FileNotFoundError:
            pass
        raise

    print(f"[migrate_labels] wrote {len(migrated)} records to {labels_path}")
    print("[migrate_labels] done. Delete this script: tools/migrate_labels.py")


def main() -> None:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--dry-run", action="store_true",
                    help="Show what would be renamed/dropped without writing")
    ap.add_argument("--apply", action="store_true",
                    help="Apply the migration (requires RENAME_MAP to be filled in)")
    ap.add_argument("--labels", default=str(DEFAULT_LABELS),
                    help="Path to labels.json")
    args = ap.parse_args()

    if args.apply and args.dry_run:
        print("[migrate_labels] ERROR: cannot use --apply and --dry-run together", file=sys.stderr)
        sys.exit(1)
    if not args.apply and not args.dry_run:
        print("[migrate_labels] ERROR: specify --dry-run or --apply", file=sys.stderr)
        sys.exit(1)

    try:
        run(Path(args.labels), dry_run=args.dry_run)
    except NotImplementedError as e:
        print(f"[migrate_labels] NotImplementedError: {e}", file=sys.stderr)
        sys.exit(2)


if __name__ == "__main__":
    main()
```

```python
"""Deterministic stub Detector and Segmenter for integration tests.

Both classes return fixed, reproducible outputs derived from the image_id
so tests can assert on exact values without GPU or model loading.
These stubs satisfy the Detector and Segmenter Protocols in interfaces.py.

Register via detectors/__init__.py and segmenters/__init__.py under "_stub".
"""
from __future__ import annotations
import hashlib

import numpy as np
from PIL import Image

from scripts.detect_subjects.interfaces import DetectionResult, SegmentationResult


class StubDetector:
    """Returns deterministic DetectionResult. No model loading."""

    model_id: str = "_stub_detector_v0"

    def __init__(self, **kwargs) -> None:
        # Accepts device/dtype kwargs from make_detector() but ignores them.
        pass

    def detect(self, image: Image.Image, image_id: str | None = None) -> DetectionResult:
        # Derive stable floats from image_id hash so outputs are reproducible.
        seed_bytes = (image_id or "default").encode()
        h = int(hashlib.sha1(seed_bytes).hexdigest()[:8], 16)
        # Normalize to [0.1, 0.4] range for bbox coords so they never touch edges
        x = 0.10 + (h % 1000) / 10000.0
        y = 0.15 + (h % 900) / 10000.0
        w = 0.20 + (h % 800) / 10000.0
        h_ = 0.25 + (h % 700) / 10000.0
        conf = 0.75 + (h % 200) / 2000.0   # [0.75, 0.85]
        ts = 0.60 + (h % 150) / 1500.0     # text score [0.60, 0.70]

        distinct = [(x, y, w, h_, conf, "a beetle")]
        return DetectionResult(
            bbox_xywh_normalized=(x, y, w, h_),
            confidence=conf,
            n_raw_detections=1,
            n_distinct_detections=1,
            detection_ms=1,
            distinct_subjects=distinct,
            text_label="a beetle",
            text_label_score=ts,
        )


class StubSegmenter:
    """Returns deterministic SegmentationResult. No model loading."""

    model_id: str = "_stub_segmenter_v0"

    def __init__(self, **kwargs) -> None:
        pass

    def segment_with_bbox(
        self,
        image_id: str,
        image: Image.Image,
        bbox_xywh_normalized: tuple[float, float, float, float],
    ) -> SegmentationResult:
        W, H = image.size
        x, y, w, h = bbox_xywh_normalized
        mask = np.zeros((H, W), dtype=bool)
        # Fill the bbox region in the mask
        x1 = int(x * W)
        y1 = int(y * H)
        x2 = int((x + w) * W)
        y2 = int((y + h) * H)
        mask[y1:y2, x1:x2] = True
        return SegmentationResult(mask=mask, iou_score=0.88, segmentation_ms=1)
```

```python
                           "\n".join(mismatches[:5]) +
```
