# Phase 3 Implementation Plan — ML Labelers + Active Learning

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Train and ship the first ML labeler (3-class blur classifier), build the active learning loop (uncertainty + diversity sampling per Encord guide), and provide a re-runnable PR curve tool for per-label threshold tuning. Grow under-represented labels (`bbox_wrong-subject`, `mask_poor-contrast`) to the 30-positive directional threshold via active labeling rounds.

**Architecture:** New `ml_labelers/` package mirrors the `detectors/` and `segmenters/` factory pattern from Phase 1. Per-label classifiers consume features from the existing `features.py` outputs. Active surfacer uses CLIP image embeddings for diversity clustering. PR curves are a standalone tool, re-runnable as new labels accumulate.

**Tech Stack:** Python 3.12, scikit-learn 1.5 (LogisticRegression), `transformers` (CLIP-ViT-B/32 for embeddings), polars 1.40, pyarrow 24.0, matplotlib (for PR curve PNGs), pytest 8.3.

**Serialization note:** Scikit-learn models are serialized with `joblib` (preferred over raw pickle for sklearn objects). The saved `.joblib` files are local artifacts trained from local data — they are never loaded from untrusted external sources. All model files live under `data/cache/ml_models/` which is gitignored.

```python
# scripts/detect_subjects/_ablation_decision.py
"""Read Phase 2b ablation outcome and return the correct feature set for blur classifier.

Phase 2b T12 writes docs/ablation_sam3_mask.json with at minimum:
  {"keep_segmenter": true|false, ...}

If keep_segmenter == true:  use 4 scalar mask features (subject_sharpness,
                            boundary_sharpness, mask_area_ratio, lab_delta_e).
If keep_segmenter == false: use CLIP image embedding of bbox crop (512-dim float).
                            Feature list contains the sentinel "clip_embedding"
                            so downstream code can branch on its presence.

All subsequent tasks import this module rather than hardcoding the branch.
"""
from __future__ import annotations
import json
from pathlib import Path

# Overrideable via monkeypatch in tests.
ABLATION_JSON_PATH: Path = (
    Path(__file__).resolve().parent.parent.parent
    / "docs"
    / "ablation_sam3_mask.json"
)

_MASK_FEATURES: list[str] = [
    "subject_sharpness",
    "boundary_sharpness",
    "mask_area_ratio",
    "lab_delta_e",
]

_CLIP_FEATURES: list[str] = ["clip_embedding"]


def get_blur_feature_set() -> list[str]:
    """Return the column names to use as blur classifier input features.

    Reads ABLATION_JSON_PATH produced by Phase 2b T12.

    Returns:
        List of feature column names.  When the list is ["clip_embedding"],
        callers must extract a 512-dim CLIP embedding from the bbox crop
        instead of reading scalar parquet columns.

    Raises:
        FileNotFoundError: ablation JSON not yet produced (run Phase 2b T12 first).
        KeyError: JSON is missing "keep_segmenter" field.
    """
    if not ABLATION_JSON_PATH.exists():
        raise FileNotFoundError(
            f"ablation JSON not found at {ABLATION_JSON_PATH!r}. "
            "Run Phase 2b T12 (ablation_sam3_mask.py) to produce it, "
            "then re-run this task."
        )
    data = json.loads(ABLATION_JSON_PATH.read_text())
    keep_segmenter: bool = data["keep_segmenter"]
    return _MASK_FEATURES if keep_segmenter else _CLIP_FEATURES


def write_branch_decision_doc(output_path: Path | None = None) -> Path:
    """Produce docs/phase3_branch_decision.md documenting the chosen branch.

    Reads ABLATION_JSON_PATH, then writes (or overwrites) the markdown doc.
    Called by train_blur_classifier.py so the doc is always current.
    """
    features = get_blur_feature_set()
    data = json.loads(ABLATION_JSON_PATH.read_text())
    keep = data["keep_segmenter"]
    branch = "mask-scalar" if keep else "clip-embedding"
    if output_path is None:
        output_path = (
            Path(__file__).resolve().parent.parent.parent
            / "docs"
            / "phase3_branch_decision.md"
        )
    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(
        f"# Phase 3 Branch Decision\n\n"
        f"**Date generated:** auto-written by `_ablation_decision.write_branch_decision_doc()`\n\n"
        f"**Ablation source:** `{ABLATION_JSON_PATH}`\n\n"
        f"**keep_segmenter:** `{keep}`\n\n"
        f"**Chosen branch:** `{branch}`\n\n"
        f"**Feature set for blur classifier:**\n"
        + "".join(f"- `{f}`\n" for f in features)
        + "\n"
        f"If `keep_segmenter=true`: blur classifier uses 4 scalar mask features.\n"
        f"If `keep_segmenter=false`: blur classifier uses 512-dim CLIP image "
        f"embedding from `openai/clip-vit-base-patch32`.\n"
    )
    return output_path
```

```python
# scripts/detect_subjects/ml_labelers/__init__.py
"""ML Labeler factory: name -> MLLabeler-implementing instance.

Mirrors scripts/detect_subjects/detectors/__init__.py.
Register new labelers by importing them here and adding to _REGISTRY.
"""
from __future__ import annotations
from typing import Any

_REGISTRY: dict[str, type] = {}


def make_ml_labeler(name: str, **kwargs: Any):
    """Construct an ML labeler by registry name."""
    if name not in _REGISTRY:
        raise ValueError(
            f"unknown ml labeler {name!r}; registered: {sorted(_REGISTRY)}"
        )
    return _REGISTRY[name](**kwargs)


def registered_ml_labelers() -> list[str]:
    return sorted(_REGISTRY)
```

```python
# scripts/detect_subjects/ml_labelers/_base.py
"""Shared training utilities for all ML labelers.

Provides:
  train_lr_classifier  -- balanced logistic regression
  train_pu_classifier  -- Elkan-Noto two-step PU learner
  cv_compare           -- picks winner per label via k-fold CV F1
"""
from __future__ import annotations
from typing import Any

import numpy as np
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import StratifiedKFold
from sklearn.metrics import f1_score


# ── LR ────────────────────────────────────────────────────────────────────────

def train_lr_classifier(
    X: np.ndarray,
    y: np.ndarray,
    class_weight: str | dict = "balanced",
    seed: int = 42,
) -> LogisticRegression:
    """Fit a balanced logistic regression on (X, y).

    Args:
        X: float32 feature matrix (n_samples, n_features).
        y: int32 binary label vector (0/1).
        class_weight: passed to LogisticRegression; "balanced" by default.
        seed: random state for reproducibility.

    Returns:
        Fitted LogisticRegression instance.
    """
    clf = LogisticRegression(
        class_weight=class_weight,
        max_iter=1000,
        random_state=seed,
        solver="lbfgs",
    )
    clf.fit(X, y)
    return clf


# ── PU ────────────────────────────────────────────────────────────────────────

class _ElkanNotoPUClassifier:
    """Minimal Elkan-Noto two-step PU classifier.

    Step 1: Train probabilistic classifier on labeled-positive vs unlabeled.
    Step 2: Estimate P(s=1|y=1) from a calibration set of labeled positives,
            then correct: P(y=1|x) = P(s=1|x) / P(s=1|y=1).

    If the `pulearn` package is available, delegates to ElkanotoPuClassifier.
    """

    def __init__(self, seed: int = 42):
        self._seed = seed
        self._base: Any = None
        self._c: float = 1.0  # P(s=1|y=1) estimate

    def fit(self, X: np.ndarray, s: np.ndarray) -> "_ElkanNotoPUClassifier":
        """Fit on (X, s) where s=1 for labeled positives, s=0 for unlabeled."""
        try:
            from pulearn import ElkanotoPuClassifier
            base = LogisticRegression(
                max_iter=1000, random_state=self._seed, solver="lbfgs"
            )
            pu = ElkanotoPuClassifier(estimator=base, hold_out_ratio=0.2)
            pu.fit(X, s)
            self._base = pu
            return self
        except ImportError:
            pass

        # Manual Elkan-Noto two-step
        rng = np.random.default_rng(self._seed)
        pos_idx = np.where(s == 1)[0]

        # Step 1: train base model on labeled-positive vs unlabeled
        base = LogisticRegression(
            class_weight="balanced", max_iter=1000,
            random_state=self._seed, solver="lbfgs"
        )
        base.fit(X, s)

        # Step 2: estimate c = P(s=1|y=1) via calibration subset of positives
        n_cal = max(1, int(0.2 * len(pos_idx)))
        cal_idx = rng.choice(pos_idx, size=n_cal, replace=False)
        p_s_given_pos = float(base.predict_proba(X[cal_idx])[:, 1].mean())
        self._c = max(p_s_given_pos, 1e-6)
        self._base = base
        return self

    def predict_proba(self, X: np.ndarray) -> np.ndarray:
        """Return corrected P(y=1|x) = P(s=1|x) / c, clipped to [0, 1]."""
        p_s = self._base.predict_proba(X)[:, 1]
        p_y = np.clip(p_s / self._c, 0.0, 1.0)
        return np.column_stack([1.0 - p_y, p_y])

    def predict(self, X: np.ndarray, threshold: float = 0.5) -> np.ndarray:
        return (self.predict_proba(X)[:, 1] >= threshold).astype(np.int32)


def train_pu_classifier(
    X: np.ndarray,
    y: np.ndarray,
    seed: int = 42,
) -> _ElkanNotoPUClassifier:
    """Fit an Elkan-Noto PU classifier.

    Args:
        X: float32 feature matrix.
        y: int32 PU labels: 1 = labeled positive, 0 = unlabeled.
        seed: random state.

    Returns:
        Fitted _ElkanNotoPUClassifier with predict() and predict_proba().
    """
    clf = _ElkanNotoPUClassifier(seed=seed)
    clf.fit(X, y)
    return clf


# ── CV compare ────────────────────────────────────────────────────────────────

def cv_compare(
    X: np.ndarray,
    y: np.ndarray,
    n_folds: int = 5,
    seed: int = 42,
) -> dict:
    """Cross-validate LR and PU classifiers; return per-model stats + winner.

    Args:
        X: float32 feature matrix.
        y: int32 binary labels (may be PU labels with unlabeled=0).
        n_folds: number of stratified CV folds.
        seed: random state.

    Returns:
        {
            "lr": {"mean_f1": float, "std_f1": float, "model": fitted_LR},
            "pu": {"mean_f1": float, "std_f1": float, "model": fitted_PU},
            "winner": "lr" | "pu",
        }
        "model" in each entry is trained on the full dataset.
    """
    skf = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=seed)
    lr_f1s: list[float] = []
    pu_f1s: list[float] = []

    for train_idx, val_idx in skf.split(X, y):
        X_tr, X_val = X[train_idx], X[val_idx]
        y_tr, y_val = y[train_idx], y[val_idx]

        lr = train_lr_classifier(X_tr, y_tr, seed=seed)
        lr_preds = lr.predict(X_val)
        lr_f1s.append(f1_score(y_val, lr_preds, zero_division=0))

        pu = train_pu_classifier(X_tr, y_tr, seed=seed)
        pu_preds = pu.predict(X_val)
        pu_f1s.append(f1_score(y_val, pu_preds, zero_division=0))

    lr_mean, lr_std = float(np.mean(lr_f1s)), float(np.std(lr_f1s))
    pu_mean, pu_std = float(np.mean(pu_f1s)), float(np.std(pu_f1s))

    winner_name = "lr" if lr_mean >= pu_mean else "pu"

    return {
        "lr": {
            "mean_f1": lr_mean,
            "std_f1": lr_std,
            "model": train_lr_classifier(X, y, seed=seed),
        },
        "pu": {
            "mean_f1": pu_mean,
            "std_f1": pu_std,
            "model": train_pu_classifier(X, y, seed=seed),
        },
        "winner": winner_name,
    }
```

```python
# scripts/detect_subjects/ml_labelers/blur.py
"""3-class blur classifier: mask_blur_unusable | mask_blur_usable | mask_good.

Implements the MLLabeler Protocol from interfaces.py.

Feature set is determined at runtime by _ablation_decision.get_blur_feature_set():
  keep_segmenter=True  -> 4 scalar mask features (subject_sharpness, boundary_sharpness,
                          mask_area_ratio, lab_delta_e)
  keep_segmenter=False -> 512-dim CLIP image embedding extracted from bbox crop

Model is loaded from MODEL_PATH (written by tools/train_blur_classifier.py using joblib).
If the model file is missing, raises FileNotFoundError with actionable message.

Note on serialization: sklearn models are saved with joblib (sklearn's recommended
serializer). Files are locally-trained artifacts; never loaded from untrusted sources.
"""
from __future__ import annotations
from pathlib import Path
from typing import Any

import joblib
import numpy as np

from scripts.detect_subjects._ablation_decision import get_blur_feature_set
from scripts.detect_subjects.config import CACHE_DIR

MODEL_PATH: Path = CACHE_DIR / "ml_models" / "blur_v0.joblib"

_CLASS_NAMES = ["mask_blur_unusable", "mask_blur_usable", "mask_good"]


class BlurClassifier:
    """3-class blur classifier implementing the MLLabeler Protocol.

    Attributes:
        model_id (str): identifier of the underlying model (e.g. "blur_v0_lr").
    """

    def __init__(self) -> None:
        if not MODEL_PATH.exists():
            raise FileNotFoundError(
                f"BlurClassifier model not found at {MODEL_PATH!r}. "
                "Run `tools/train_blur_classifier.py` first to train and save the model:\n"
                "  .venv/bin/python -m tools.train_blur_classifier"
            )
        payload: dict = joblib.load(MODEL_PATH)
        self._model: Any = payload["model"]
        self.model_id: str = payload.get("model_id", "blur_v0_lr")
        self._feature_set: list[str] = payload.get(
            "feature_set", get_blur_feature_set()
        )
        self._classes: list[str] = payload.get("classes", _CLASS_NAMES)

    def _build_feature_vector(self, features: dict) -> np.ndarray:
        """Assemble 1-row feature matrix from features dict.

        For scalar branch: reads named columns in order.
        For CLIP branch: expects features["clip_embedding"] (512-dim array).
        """
        if self._feature_set == ["clip_embedding"]:
            emb = np.asarray(features["clip_embedding"], dtype=np.float32)
            return emb.reshape(1, -1)
        vec = np.array(
            [float(features.get(col) or 0.0) for col in self._feature_set],
            dtype=np.float32,
        )
        return vec.reshape(1, -1)

    def predict(self, image_id: str, features: dict) -> dict[str, float]:
        """Return class probabilities for blur quality labels.

        Args:
            image_id: identifier for the image (used for logging upstream).
            features: dict of feature values; keys depend on chosen feature branch.

        Returns:
            {"mask_blur_unusable": float, "mask_blur_usable": float, "mask_good": float}
            Values are probabilities summing to 1.0.
        """
        X = self._build_feature_vector(features)
        proba = self._model.predict_proba(X)[0]
        return {cls: float(p) for cls, p in zip(self._classes, proba)}
```

```python
# tools/train_blur_classifier.py
"""Train the 3-class blur classifier and save it to data/cache/ml_models/blur_v0.joblib.

Usage:
    .venv/bin/python -m tools.train_blur_classifier

Reads:
    data/cache/framing_detections.parquet  -- detection features (SAM 3 variant)
    data/cache/labels.json                 -- human blur labels

Writes:
    data/cache/ml_models/blur_v0.joblib   -- model payload via joblib (gitignored)
    docs/ml/blur_v0_cv_report.md           -- CV results (committed)

Note: joblib is sklearn's recommended serializer for trained estimators.
The model file is a local artifact trained from local data.
"""
from __future__ import annotations
import json
from pathlib import Path

import joblib
import numpy as np
import pyarrow.parquet as pq

from scripts.detect_subjects._ablation_decision import get_blur_feature_set, write_branch_decision_doc
from scripts.detect_subjects.ml_labelers._base import cv_compare

# ── Configurable paths (patchable in tests) ────────────────────────────────

ROOT = Path(__file__).resolve().parent.parent
PARQUET_PATH: Path = ROOT / "data" / "cache" / "framing_detections.parquet"
LABELS_PATH: Path = ROOT / "data" / "cache" / "labels.json"
MODEL_OUTPUT_PATH: Path = ROOT / "data" / "cache" / "ml_models" / "blur_v0.joblib"
CV_REPORT_PATH: Path = ROOT / "docs" / "ml" / "blur_v0_cv_report.md"
VARIANT_FILTER: str = "v2_dino_sam3"

_CLASS_NAMES = ["mask_blur_unusable", "mask_blur_usable", "mask_good"]


def _load_parquet(parquet_path: Path, variant: str) -> dict[str, dict]:
    """Load parquet rows filtered to variant; return {image_id: feature_dict}."""
    table = pq.read_table(parquet_path)
    df = table.to_pydict()
    n = len(df["image_id"])
    rows: dict[str, dict] = {}
    for i in range(n):
        if df.get("variant", [None] * n)[i] != variant:
            continue
        iid = df["image_id"][i]
        rows[iid] = {col: df[col][i] for col in df if col not in ("image_id", "variant")}
    return rows


def _load_labels(labels_path: Path) -> dict[str, str]:
    """Load labels.json; return {image_id: blur_class_name} for labeled images."""
    raw = json.loads(labels_path.read_text())
    labeled: dict[str, str] = {}
    for iid, rec in raw.items():
        mask_labels = rec.get("mask_labels", [])
        for cls in _CLASS_NAMES:
            if cls in mask_labels:
                labeled[iid] = cls
                break
    return labeled


def _build_xy(
    feature_rows: dict[str, dict],
    labels: dict[str, str],
    feature_set: list[str],
) -> tuple[np.ndarray, np.ndarray, list[str]]:
    """Build X, y arrays from matched rows + labels."""
    image_ids = sorted(set(feature_rows) & set(labels))
    cls_to_idx = {c: i for i, c in enumerate(_CLASS_NAMES)}
    X_rows = []
    y_rows = []
    for iid in image_ids:
        if feature_set == ["clip_embedding"]:
            emb = feature_rows[iid].get("clip_embedding")
            if emb is None:
                continue
            X_rows.append(np.asarray(emb, dtype=np.float32))
        else:
            vec = [float(feature_rows[iid].get(col) or 0.0) for col in feature_set]
            X_rows.append(np.array(vec, dtype=np.float32))
        y_rows.append(cls_to_idx[labels[iid]])
    if not X_rows:
        raise RuntimeError("No labeled+featured images found.")
    return np.stack(X_rows), np.array(y_rows, dtype=np.int32), image_ids


def _write_cv_report(
    cv_path: Path,
    lr_mean: float, lr_std: float,
    pu_mean: float, pu_std: float,
    winner: str,
    n_samples: int,
    feature_set: list[str],
) -> None:
    cv_path.parent.mkdir(parents=True, exist_ok=True)
    cv_path.write_text(
        "# Blur Classifier v0 — CV Report\n\n"
        f"**n_samples:** {n_samples}  \n"
        f"**feature_set:** {feature_set}  \n"
        f"**winner:** `{winner}`\n\n"
        "| Model | Mean F1 | Std F1 |\n"
        "|-------|---------|--------|\n"
        f"| LR (balanced) | {lr_mean:.4f} | {lr_std:.4f} |\n"
        f"| PU (Elkan-Noto) | {pu_mean:.4f} | {pu_std:.4f} |\n"
    )
    print(f"[train_blur] CV report written -> {cv_path}")


def train() -> None:
    """Main training entry point."""
    feature_set = get_blur_feature_set()
    print(f"[train_blur] feature_set = {feature_set}")

    feature_rows = _load_parquet(PARQUET_PATH, VARIANT_FILTER)
    print(f"[train_blur] {len(feature_rows)} feature rows")

    labels = _load_labels(LABELS_PATH)
    print(f"[train_blur] {len(labels)} blur-labeled images")

    X, y, image_ids = _build_xy(feature_rows, labels, feature_set)
    print(f"[train_blur] training matrix: {X.shape}")

    result = cv_compare(X, y, n_folds=5, seed=42)
    winner = result["winner"]
    model = result[winner]["model"]
    model_id = f"blur_v0_{winner}"
    print(f"[train_blur] winner={winner} LR={result['lr']['mean_f1']:.3f} PU={result['pu']['mean_f1']:.3f}")

    MODEL_OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    payload = {
        "model": model,
        "model_id": model_id,
        "feature_set": feature_set,
        "classes": _CLASS_NAMES,
    }
    joblib.dump(payload, MODEL_OUTPUT_PATH)
    print(f"[train_blur] model saved -> {MODEL_OUTPUT_PATH}")

    _write_cv_report(
        CV_REPORT_PATH,
        lr_mean=result["lr"]["mean_f1"],
        lr_std=result["lr"]["std_f1"],
        pu_mean=result["pu"]["mean_f1"],
        pu_std=result["pu"]["std_f1"],
        winner=winner,
        n_samples=len(image_ids),
        feature_set=feature_set,
    )
    write_branch_decision_doc()
    print("[train_blur] done.")


if __name__ == "__main__":
    train()
```

```python
# scripts/detect_subjects/pr_curves.py
"""Standalone PR curve tool -- re-runnable as new labels accumulate.

CLI:
    .venv/bin/python -m scripts.detect_subjects.pr_curves \\
        --label mask_blur_unusable \\
        --thresholds 0.01,0.05,0.10,0.20,0.30,0.40,0.50,0.60,0.70,0.80,0.90,0.95,0.99

For the given label:
  - Reads ml_proba__<label> column from parquet
  - Reads ground-truth from labels.json
  - Prints markdown table (threshold | precision | recall | f1) to stdout
  - Writes PNG to docs/pr_curves/<date>/<label>.png  (gitignored directory)
  - Writes CSV alongside PNG for re-analysis
"""
from __future__ import annotations
import argparse
import csv
import datetime
import json
from pathlib import Path
from typing import Optional

import numpy as np
import pyarrow.parquet as pq

from scripts.detect_subjects.config import CACHE_DIR

PARQUET_PATH: Path = CACHE_DIR / "framing_detections.parquet"
LABELS_PATH: Path = CACHE_DIR / "labels.json"
ROOT = Path(__file__).resolve().parent.parent.parent
DEFAULT_OUTPUT_DIR = ROOT / "docs" / "pr_curves"


def _pr_at_threshold(
    y_true: np.ndarray, y_proba: np.ndarray, threshold: float
) -> tuple[float, float, float]:
    """Return (precision, recall, f1) at one threshold."""
    preds = (y_proba >= threshold).astype(int)
    tp = int(((preds == 1) & (y_true == 1)).sum())
    fp = int(((preds == 1) & (y_true == 0)).sum())
    fn = int(((preds == 0) & (y_true == 1)).sum())
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall    = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = (2 * precision * recall / (precision + recall)
          if (precision + recall) > 0 else 0.0)
    return precision, recall, f1


def run(
    label: str,
    thresholds: list[float],
    output_dir: Optional[Path] = None,
    save_png: bool = True,
) -> list[dict]:
    """Compute PR curve for `label`, print table, write CSV + optional PNG.

    Args:
        label: ml_proba__ column suffix (e.g. "mask_blur_unusable").
        thresholds: decision thresholds to evaluate.
        output_dir: directory for CSV + PNG; defaults to docs/pr_curves/<date>/
        save_png: set False to skip matplotlib (CI / headless).

    Returns:
        List of dicts: [{threshold, precision, recall, f1}, ...]

    Raises:
        ValueError: parquet has no ml_proba__<label> column.
    """
    proba_col = f"ml_proba__{label}"

    table = pq.read_table(PARQUET_PATH)
    col_names = table.schema.names
    if proba_col not in col_names:
        raise ValueError(
            f"Column {proba_col!r} not found in parquet. "
            f"Available ml_proba__ columns: "
            f"{[c for c in col_names if c.startswith('ml_proba__')]}"
        )

    image_ids = table.column("image_id").to_pylist()
    raw_probas = table.column(proba_col).to_pylist()
    labels_json: dict = json.loads(LABELS_PATH.read_text())

    y_true_list, y_proba_list = [], []
    for iid, proba in zip(image_ids, raw_probas):
        if proba is None:
            continue
        rec = labels_json.get(iid)
        if rec is None:
            continue
        gt = 1 if label in rec.get("mask_labels", []) else 0
        y_true_list.append(gt)
        y_proba_list.append(float(proba))

    y_true  = np.array(y_true_list,  dtype=np.int32)
    y_proba = np.array(y_proba_list, dtype=np.float64)

    n_pos = int(y_true.sum())
    print(f"\n## PR Curve -- {label}")
    print(f"n_labeled={len(y_true)}  n_positive={n_pos}  n_negative={len(y_true) - n_pos}\n")

    rows: list[dict] = []
    for t in sorted(thresholds):
        p, r, f = _pr_at_threshold(y_true, y_proba, t)
        rows.append({"threshold": t, "precision": p, "recall": r, "f1": f})

    print("| Threshold | Precision | Recall | F1    |")
    print("|-----------|-----------|--------|-------|")
    for row in rows:
        print(f"| {row['threshold']:.2f}      | {row['precision']:.3f}     "
              f"| {row['recall']:.3f}  | {row['f1']:.3f} |")
    print()

    if output_dir is None:
        output_dir = DEFAULT_OUTPUT_DIR / datetime.date.today().isoformat()
    output_dir.mkdir(parents=True, exist_ok=True)

    csv_path = output_dir / f"{label}.csv"
    with csv_path.open("w", newline="") as fh:
        writer = csv.DictWriter(fh, fieldnames=["threshold", "precision", "recall", "f1"])
        writer.writeheader()
        writer.writerows(rows)
    print(f"[pr_curves] CSV -> {csv_path}")

    if save_png:
        try:
            import matplotlib
            matplotlib.use("Agg")
            import matplotlib.pyplot as plt

            fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
            ax1.plot([r["recall"] for r in rows], [r["precision"] for r in rows],
                     "b-o", label="PR")
            ax1.set(xlabel="Recall", ylabel="Precision",
                    title=f"PR Curve -- {label}", xlim=(0, 1), ylim=(0, 1))
            ax2.plot([r["threshold"] for r in rows], [r["f1"] for r in rows],
                     "g-s", label="F1")
            ax2.set(xlabel="Threshold", ylabel="F1",
                    title=f"F1 vs Threshold -- {label}", xlim=(0, 1), ylim=(0, 1))
            fig.tight_layout()
            png_path = output_dir / f"{label}.png"
            fig.savefig(png_path, dpi=150)
            plt.close(fig)
            print(f"[pr_curves] PNG -> {png_path}")
        except Exception as exc:
            print(f"[pr_curves] WARNING: could not save PNG: {exc}")

    return rows


def main() -> None:
    parser = argparse.ArgumentParser(
        description="Compute PR curves for ml_proba__<label> columns."
    )
    parser.add_argument("--label", required=True)
    parser.add_argument(
        "--thresholds",
        default="0.01,0.05,0.10,0.20,0.30,0.40,0.50,0.60,0.70,0.80,0.90,0.95,0.99",
    )
    args = parser.parse_args()
    thresholds = [float(x.strip()) for x in args.thresholds.split(",") if x.strip()]
    run(label=args.label, thresholds=thresholds)


if __name__ == "__main__":
    main()
```

```python
# tools/migrate_parquet_v3.py
"""One-shot migration: add ml_proba__ columns to existing parquet with None defaults.

Run ONCE, verify, then delete this file per repo convention:
    git rm tools/migrate_parquet_v3.py

Usage:
    .venv/bin/python -m tools.migrate_parquet_v3
"""
from __future__ import annotations
import shutil
from pathlib import Path

import pyarrow as pa
import pyarrow.parquet as pq

from scripts.detect_subjects.config import CACHE_DIR

PARQUET_PATH = CACHE_DIR / "framing_detections.parquet"
BACKUP_PATH  = CACHE_DIR / "framing_detections.parquet.pre_v3_backup"

_NEW_COLS = [
    ("ml_proba__mask_blur_unusable", pa.float32()),
    ("ml_proba__mask_blur_usable",   pa.float32()),
    ("ml_proba__mask_good",          pa.float32()),
]


def migrate() -> None:
    if not PARQUET_PATH.exists():
        print(f"[migrate_v3] Nothing to migrate -- {PARQUET_PATH} not found.")
        return

    table = pq.read_table(PARQUET_PATH)
    existing = set(table.schema.names)
    if all(col in existing for col, _ in _NEW_COLS):
        print("[migrate_v3] All ml_proba__ columns already present. Nothing to do.")
        return

    shutil.copy2(PARQUET_PATH, BACKUP_PATH)
    print(f"[migrate_v3] Backup -> {BACKUP_PATH}")

    n = table.num_rows
    for col_name, col_type in _NEW_COLS:
        if col_name not in existing:
            table = table.append_column(
                pa.field(col_name, col_type),
                pa.array([None] * n, type=col_type),
            )
            print(f"[migrate_v3] Added {col_name!r}")

    pq.write_table(table, PARQUET_PATH, compression="snappy")
    print(f"[migrate_v3] Done -> {PARQUET_PATH}")
    print(f"[migrate_v3] Verify OK, then:")
    print(f"  rm {BACKUP_PATH}")
    print(f"  git rm tools/migrate_parquet_v3.py")


if __name__ == "__main__":
    migrate()
```

```python
"""Active learning uncertainty+diversity sampler for the framing pipeline.

surface_candidates():
  1. Reads parquet + labels.json to find unlabeled/unreviewed images.
  2. Reads ml_proba__<label> column; skips rows where it is None.
  3. Filters to the uncertain band: confidence in [0.3, 0.7].
  4. Clusters uncertain images by CLIP image embedding (KMeans).
  5. Returns one representative per cluster (closest to centroid), up to n_per_round.
  6. Writes manifest to data/cache/active_surfacing.json.
"""
from __future__ import annotations
import json
import logging
import os
import tempfile
from pathlib import Path

import numpy as np
import polars as pl

from scripts.detect_subjects.config import CACHE_DIR, IMG_DIR, PARQUET_PATH

log = logging.getLogger(__name__)

LABELS_PATH = CACHE_DIR / "labels.json"
ACTIVE_SURFACING_PATH = CACHE_DIR / "active_surfacing.json"
CLIP_MODEL_ID = "openai/clip-vit-base-patch32"
UNCERTAIN_LOW = 0.3
UNCERTAIN_HIGH = 0.7

_CLIP_MODEL = None
_CLIP_PROCESSOR = None


def _ensure_clip() -> tuple:
    global _CLIP_MODEL, _CLIP_PROCESSOR
    if _CLIP_MODEL is None:
        from transformers import CLIPModel, CLIPProcessor
        log.info("[active_surfacer] loading CLIP model %s", CLIP_MODEL_ID)
        _CLIP_PROCESSOR = CLIPProcessor.from_pretrained(CLIP_MODEL_ID)
        _CLIP_MODEL = CLIPModel.from_pretrained(CLIP_MODEL_ID)
        _CLIP_MODEL.eval()
    return _CLIP_MODEL, _CLIP_PROCESSOR


def _load_images_for_ids(image_ids: list[str], img_dir: Path) -> list:
    from PIL import Image
    images = []
    for iid in image_ids:
        for ext in ("jpg", "jpeg", "png"):
            p = img_dir / f"{iid}.{ext}"
            if p.exists():
                try:
                    with Image.open(p) as im:
                        images.append(im.convert("RGB").copy())
                except Exception:
                    images.append(None)
                break
        else:
            images.append(None)
    return images


def _get_clip_embeddings(pil_images: list, batch_size: int = 32) -> np.ndarray:
    import torch
    model, processor = _ensure_clip()
    all_embs: list[np.ndarray] = []
    for i in range(0, len(pil_images), batch_size):
        batch = pil_images[i : i + batch_size]
        valid = [(j, img) for j, img in enumerate(batch) if img is not None]
        chunk = np.zeros((len(batch), 512), dtype=np.float32)
        if valid:
            idxs, imgs = zip(*valid)
            inputs = processor(images=list(imgs), return_tensors="pt")
            with torch.no_grad():
                feats = model.get_image_features(**inputs)
            feats_np = feats.cpu().numpy().astype(np.float32)
            for k, idx in enumerate(idxs):
                chunk[idx] = feats_np[k]
        all_embs.append(chunk)
    return np.concatenate(all_embs, axis=0) if all_embs else np.zeros((0, 512), dtype=np.float32)


def _atomic_write_json(path: Path, data: dict | list) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    fd, tmp = tempfile.mkstemp(prefix=".tmp_", dir=str(path.parent), suffix=".json")
    try:
        with os.fdopen(fd, "w") as f:
            json.dump(data, f, indent=2)
        os.replace(tmp, path)
    except Exception:
        try:
            os.unlink(tmp)
        except FileNotFoundError:
            pass
        raise


def surface_candidates(
    label: str,
    n_per_round: int = 25,
    n_clusters: int = 5,
    parquet_path: Path = PARQUET_PATH,
    labels_path: Path = LABELS_PATH,
    cache_dir: Path = CACHE_DIR,
    img_dir: Path = IMG_DIR,
    pr_curves_root: Path | None = None,
) -> list[dict]:
    """Sample uncertain+diverse candidates for `label` for the next active learning round.

    Returns a list of dicts: {image_id, confidence, cluster_id, surfacing_reason}.
    Also writes/updates data/cache/active_surfacing.json keyed by label name.
    """
    from sklearn.cluster import KMeans

    if pr_curves_root is None:
        pr_curves_root = Path(__file__).resolve().parents[2] / "docs" / "pr_curves"

    # Stop criterion check.
    if _check_stop_criterion(label, pr_curves_root):
        return []

    df = pl.read_parquet(parquet_path)
    col_name = f"ml_proba__{label}"
    if col_name not in df.columns:
        log.warning("[active_surfacer] column %s not in parquet; returning empty", col_name)
        return []

    labels: dict = {}
    if labels_path.exists():
        try:
            labels = json.loads(labels_path.read_text() or "{}")
        except Exception:
            labels = {}

    reviewed_ids: set[str] = {
        iid for iid, rec in labels.items()
        if rec and rec.get("reviewed_at") is not None
    }

    df_cands = (
        df
        .filter(pl.col(col_name).is_not_null())
        .filter(~pl.col("image_id").is_in(list(reviewed_ids)))
    )
    if df_cands.height == 0:
        log.info("[active_surfacer] no unreviewed candidates with %s populated", col_name)
        return []

    df_uncertain = df_cands.filter(
        (pl.col(col_name) >= UNCERTAIN_LOW) & (pl.col(col_name) <= UNCERTAIN_HIGH)
    )
    if df_uncertain.height == 0:
        log.info("[active_surfacer] no uncertain candidates in [%.1f, %.1f]", UNCERTAIN_LOW, UNCERTAIN_HIGH)
        return []

    uncertain_ids = df_uncertain["image_id"].to_list()
    uncertain_probas = df_uncertain[col_name].to_list()

    pil_images = _load_images_for_ids(uncertain_ids, img_dir)
    embeddings = _get_clip_embeddings(pil_images)

    actual_k = min(n_clusters, len(uncertain_ids))
    kmeans = KMeans(n_clusters=actual_k, random_state=42, n_init="auto")
    cluster_labels = kmeans.fit_predict(embeddings)

    selected: list[dict] = []
    for cid in range(actual_k):
        members = [
            (i, uncertain_ids[i], uncertain_probas[i])
            for i in range(len(uncertain_ids))
            if cluster_labels[i] == cid
        ]
        if not members:
            continue
        centroid = kmeans.cluster_centers_[cid]
        best_i, best_id, best_proba = min(
            members,
            key=lambda m: float(np.linalg.norm(embeddings[m[0]] - centroid)),
        )
        selected.append({
            "image_id": best_id,
            "confidence": float(best_proba),
            "cluster_id": int(cid),
            "surfacing_reason": f"uncertain ({best_proba:.2f}), cluster {cid}",
        })
        if len(selected) >= n_per_round:
            break

    manifest_path = cache_dir / "active_surfacing.json"
    manifest: dict = {}
    if manifest_path.exists():
        try:
            manifest = json.loads(manifest_path.read_text() or "{}")
        except Exception:
            manifest = {}
    manifest[label] = selected
    _atomic_write_json(manifest_path, manifest)

    log.info("[active_surfacer] surfaced %d candidates for label=%s", len(selected), label)
    return selected


# ── Stop criterion (also used by Task 10) ───────────────────────────

import csv

STOP_MIN_ROUNDS = 3
STOP_F1_DELTA_THRESHOLD = 0.02


def _check_stop_criterion(label: str, pr_curves_root: Path) -> bool:
    """Return True if the last 3 PR curve runs show < 0.02 F1 change.

    Looks for dated subdirectories under pr_curves_root.
    Each directory must contain a CSV named <label>.csv with a 'f1' column.
    Returns False (proceed) when fewer than 3 runs exist or the root is absent.
    """
    if not pr_curves_root.exists():
        return False

    dated_dirs = sorted(
        [d for d in pr_curves_root.iterdir()
         if d.is_dir() and (d / f"{label}.csv").exists()],
        key=lambda d: d.name,
    )
    if len(dated_dirs) < STOP_MIN_ROUNDS:
        return False

    recent = dated_dirs[-STOP_MIN_ROUNDS:]
    max_f1_per_round: list[float] = []
    for d in recent:
        csv_path = d / f"{label}.csv"
        try:
            with csv_path.open() as f:
                reader = csv.DictReader(f)
                f1_vals = [float(row["f1"]) for row in reader if row.get("f1")]
            max_f1_per_round.append(max(f1_vals) if f1_vals else 0.0)
        except Exception as e:
            log.warning("[active_surfacer] could not parse %s: %s", csv_path, e)
            return False

    overall_max = max(max_f1_per_round)
    overall_min = min(max_f1_per_round)
    if (overall_max - overall_min) < STOP_F1_DELTA_THRESHOLD:
        log.info(
            "[active_surfacer] label %s appears stable across %d rounds "
            "(max F1 range: %.3f); skipping surfacing",
            label, STOP_MIN_ROUNDS, overall_max - overall_min,
        )
        return True
    return False
```

```python
"""Active learning round orchestrator.

Sequential steps per round:
  1. Re-train all registered ML classifiers.
  2. Re-run ML labeler stage to refresh ml_proba__* columns in parquet.
  3. Generate active surfacing manifest via active_surfacer.surface_candidates.
  4. Print summary + user instructions.
  5. Append round record to data/cache/active_rounds.json.

Usage:
    .venv/bin/python -m scripts.detect_subjects.run_active_round
"""
from __future__ import annotations
import json
import logging
import os
import subprocess
import sys
import tempfile
import time
from pathlib import Path

from scripts.detect_subjects.active_surfacer import surface_candidates
from scripts.detect_subjects.config import CACHE_DIR

log = logging.getLogger(__name__)

ROUNDS_LOG_PATH = CACHE_DIR / "active_rounds.json"
ACTIVE_SURFACING_PATH = CACHE_DIR / "active_surfacing.json"
VALIDATOR_URL = "http://localhost:8765"
VALIDATOR_HTML = "audit/framing-validator/v1_dino_insectsam.html"


def registered_ml_labelers() -> list[str]:
    try:
        from scripts.detect_subjects.ml_labelers import registered_ml_labelers as _reg
        return _reg()
    except ImportError:
        log.warning("[run_active_round] ml_labelers package not found; no labelers registered")
        return []


def _retrain_all_classifiers() -> None:
    log.info("[run_active_round] retraining all classifiers")
    scripts_root = Path(__file__).resolve().parents[2]
    detect_dir = scripts_root / "scripts" / "detect_subjects"
    train_scripts = sorted(detect_dir.glob("train_*_classifier.py"))
    if not train_scripts:
        log.info("[run_active_round] no train_*_classifier.py scripts found; skipping retrain")
        return
    for script in train_scripts:
        log.info("[run_active_round] running %s", script.name)
        result = subprocess.run([sys.executable, str(script)], capture_output=False, check=False)
        if result.returncode != 0:
            log.warning("[run_active_round] %s exited with code %d", script.name, result.returncode)


def _repopulate_ml_proba_columns() -> None:
    log.info("[run_active_round] repopulating ml_proba columns")
    try:
        from scripts.detect_subjects.ml_labelers import registered_ml_labelers as _reg
        labels = _reg()
        if not labels:
            log.info("[run_active_round] no ML labelers registered; nothing to repopulate")
            return
        log.info(
            "[run_active_round] ML labelers registered: %s. "
            "Re-run classify.py to refresh parquet ml_proba__ columns with the new model.",
            labels,
        )
    except ImportError:
        log.warning("[run_active_round] ml_labelers package not available; parquet not updated")


def _atomic_write_json(path: Path, data: dict | list) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    fd, tmp = tempfile.mkstemp(prefix=".tmp_", dir=str(path.parent), suffix=".json")
    try:
        with os.fdopen(fd, "w") as f:
            json.dump(data, f, indent=2)
        os.replace(tmp, path)
    except Exception:
        try:
            os.unlink(tmp)
        except FileNotFoundError:
            pass
        raise


def _execute_round(
    rounds_log_path: Path = ROUNDS_LOG_PATH,
    active_surfacing_path: Path = ACTIVE_SURFACING_PATH,
) -> dict:
    _retrain_all_classifiers()
    _repopulate_ml_proba_columns()

    labels = registered_ml_labelers()
    labels_surfaced: dict[str, int] = {}
    for label in labels:
        candidates = surface_candidates(label=label)
        labels_surfaced[label] = len(candidates)

    existing_rounds: list[dict] = []
    if rounds_log_path.exists():
        try:
            existing_rounds = json.loads(rounds_log_path.read_text() or "[]")
        except Exception:
            existing_rounds = []
    round_id = (max((r["round_id"] for r in existing_rounds), default=0)) + 1

    entry = {
        "round_id": round_id,
        "timestamp_ms": int(time.time() * 1000),
        "labels_surfaced": labels_surfaced,
    }
    existing_rounds.append(entry)
    _atomic_write_json(rounds_log_path, existing_rounds)

    print(f"\n=== Active Learning Round {round_id} Complete ===")
    print("Candidates surfaced per label:")
    if labels_surfaced:
        for lbl, n in labels_surfaced.items():
            print(f"  {lbl}: {n}")
    else:
        print("  (no registered labels yet)")
    total = sum(labels_surfaced.values())
    print(f"\nTotal: {total} candidates")
    print(
        f"\nRound ready. Open {VALIDATOR_URL}/{VALIDATOR_HTML}, "
        "select 'Active surfacing queue' filter, label the surfaced images per label. "
        "When done, run scripts/detect_subjects/run_active_round.py again."
    )

    return entry


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO, format="%(message)s")
    _execute_round()
```

```python
"""PoorContrastClassifier — binary ML labeler for mask_poor-contrast.

Features: mask_area_ratio, lab_delta_e, boundary_sharpness, mask_iou_score
(mask features; mirrors BlurClassifier feature selection).
"""
from __future__ import annotations
import logging
from pathlib import Path

import numpy as np

log = logging.getLogger(__name__)

POSITIVE_LABEL = "mask_poor-contrast"
NEGATIVE_LABEL = "mask_good_contrast"
MODEL_PATH = Path(__file__).resolve().parents[3] / "data" / "cache" / "poor_contrast_classifier.pkl"

_MASK_FEATURES = [
    "mask_area_ratio",
    "lab_delta_e",
    "boundary_sharpness",
    "mask_iou_score",
]


class PoorContrastClassifier:
    """Binary classifier for poor-contrast detection. MLLabeler protocol."""

    def __init__(self, model_path: Path = MODEL_PATH) -> None:
        import joblib
        self._model = joblib.load(model_path)
        log.info("[poor_contrast] loaded model from %s", model_path)

    def _build_feature_vector(self, features: dict) -> np.ndarray:
        vals = [float(features.get(k) or 0.0) for k in _MASK_FEATURES]
        return np.array(vals, dtype=np.float32).reshape(1, -1)

    def predict(self, image_id: str, features: dict) -> dict[str, float]:
        X = self._build_feature_vector(features)
        proba_matrix = self._model.predict_proba(X)
        classes = list(self._model.classes_)
        proba_dict = {cls: float(proba_matrix[0, i]) for i, cls in enumerate(classes)}
        pos = proba_dict.get(POSITIVE_LABEL, 0.0)
        neg = proba_dict.get(NEGATIVE_LABEL, 1.0 - pos)
        return {POSITIVE_LABEL: pos, NEGATIVE_LABEL: neg}
```

```python
"""Train PoorContrastClassifier and save to data/cache/poor_contrast_classifier.pkl.

Usage:
    .venv/bin/python -m scripts.detect_subjects.train_poor_contrast_classifier
"""
from __future__ import annotations
import json
import logging
import sys
from pathlib import Path

import joblib
import numpy as np
import polars as pl

from scripts.detect_subjects.config import CACHE_DIR, PARQUET_PATH
from scripts.detect_subjects.ml_labelers._base import cv_compare, train_lr_classifier
from scripts.detect_subjects.ml_labelers.poor_contrast import (
    MODEL_PATH, NEGATIVE_LABEL, POSITIVE_LABEL, _MASK_FEATURES,
)

log = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO, format="%(message)s")

LABELS_PATH = CACHE_DIR / "labels.json"
PREF_LABEL_FROM_KIND = {"original": "original-good", "cropped": "cropped-good"}


def _load_labeled_rows() -> tuple[np.ndarray, np.ndarray]:
    if not LABELS_PATH.exists():
        raise FileNotFoundError(f"labels.json not found at {LABELS_PATH}")
    labels: dict = json.loads(LABELS_PATH.read_text())
    df = pl.read_parquet(PARQUET_PATH)
    X_rows: list[list[float]] = []
    y_vals: list[int] = []
    for row in df.iter_rows(named=True):
        iid = row["image_id"]
        rec = labels.get(iid)
        if not rec or not rec.get("reviewed_at") or rec.get("unsure"):
            continue
        flags: set[str] = set(rec.get("flags") or [])
        pref = rec.get("preference")
        if pref in PREF_LABEL_FROM_KIND:
            flags.add(PREF_LABEL_FROM_KIND[pref])
        is_positive = "poor-contrast" in flags
        feats = [float(row.get(f) or 0.0) for f in _MASK_FEATURES]
        if all(v == 0.0 for v in feats):
            continue
        X_rows.append(feats)
        y_vals.append(1 if is_positive else 0)
    if not X_rows:
        raise RuntimeError("no labeled rows with mask features found; cannot train")
    return np.array(X_rows, dtype=np.float32), np.array(y_vals, dtype=int)


def main() -> None:
    log.info("[train_poor_contrast] loading labeled data")
    X, y = _load_labeled_rows()
    pos = int(y.sum())
    log.info("[train_poor_contrast] %d samples (%d positive, %d negative)", len(y), pos, len(y) - pos)
    model = train_lr_classifier(X, y, class_names=[NEGATIVE_LABEL, POSITIVE_LABEL])
    metrics = cv_compare(X, y, model)
    log.info("[train_poor_contrast] CV F1 = %.3f +/- %.3f", metrics["mean_f1"], metrics["std_f1"])
    MODEL_PATH.parent.mkdir(parents=True, exist_ok=True)
    joblib.dump(model, MODEL_PATH)
    log.info("[train_poor_contrast] model saved to %s", MODEL_PATH)


if __name__ == "__main__":
    main()
```
