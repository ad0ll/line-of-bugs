**Goal:** Calibrate the drawability gate to hit >= 0.94 precision per parent spec, run the full pipeline over all 34k raw images, sample-review the keep/reject output (50 keeps + 50 rejects), and decide whether to ship to gallery.

**Architecture:** Phase 4 is sequential — each step depends on the previous step's outcome. Gate sweep first (picks per-label thresholds for precision target). Lock thresholds. Run pipeline over 34k with checkpoint/resume. Random-sample 100 outputs. Manual review via reuse-existing-validator-UI in a "sample-review mode." Disagreement count drives ship-vs-iterate decision tree.

**Tech Stack:** Python 3.12, polars 1.40, pyarrow 24.0, scikit-learn 1.5 (for ML labeler thresholding), pytest 8.3. Reuses the validator UI from Phase 2a.

```python
#!/usr/bin/env python3
"""Measure classify.py throughput on the existing labeled subset.

Reads the current parquet (filtered to active SAM 3 variant), picks up to
N_SAMPLE rows that have already been processed, and times a re-run over a
fresh temp parquet to avoid cache hits. Outputs wall-clock + per-image rates
and a 34k projection. If projection > BUDGET_HOURS, prints a WARNING flag.

Usage:
    .venv/bin/python tools/measure_throughput.py [--sample N] [--out docs/phase4_throughput.md]
"""
from __future__ import annotations
import argparse
import sys
import time
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))

import polars as pl
import sqlite3
import tempfile

from scripts.detect_subjects.config import (
    DATA_DIR,
    PARQUET_PATH,
    RANDOM_SEED,
)
from scripts.detect_subjects import config as cfg
from scripts.detect_subjects.classify import run_v1_on_sample

FULL_DATASET_SIZE = 34_000
BUDGET_HOURS = 4.0


def _load_sample_rows(parquet_path: Path, variant: str, n: int) -> list[dict]:
    """Return up to n DB rows that already have parquet entries for variant."""
    if not parquet_path.exists():
        raise FileNotFoundError(f"parquet not found: {parquet_path}")
    df = pl.read_parquet(parquet_path).filter(pl.col("variant") == variant)
    if df.height == 0:
        raise RuntimeError(f"no rows for variant={variant} in {parquet_path}")
    sample = df.sample(min(n, df.height), seed=RANDOM_SEED)
    image_ids = set(sample["image_id"].to_list())

    db_path = DATA_DIR / "db" / "line-of-bugs.db"
    con = sqlite3.connect(str(db_path))
    try:
        placeholders = ",".join("?" * len(image_ids))
        rows = con.execute(
            f"SELECT image_id, filename, source FROM images "
            f"WHERE image_id IN ({placeholders}) AND filename IS NOT NULL",
            list(image_ids),
        ).fetchall()
    finally:
        con.close()
    return [{"image_id": r[0], "filename": r[1], "source": r[2]} for r in rows]


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--sample", type=int, default=50,
                    help="Number of images to time (default: 50)")
    ap.add_argument("--out", default="docs/phase4_throughput.md",
                    help="Output markdown path (default: docs/phase4_throughput.md)")
    args = ap.parse_args()

    variant = cfg.variant_tag()
    sample_rows = _load_sample_rows(PARQUET_PATH, variant, args.sample)
    if not sample_rows:
        print("ERROR: no processable rows found -- is the parquet populated?")
        sys.exit(1)
    actual_n = len(sample_rows)
    print(f"[throughput] timing {actual_n} images (variant={variant})")

    with tempfile.NamedTemporaryFile(suffix=".parquet", delete=False) as f:
        tmp_parquet = Path(f.name)

    try:
        t0 = time.perf_counter()
        summary = run_v1_on_sample(
            sample_rows,
            gt_index=None,
            parquet_path=tmp_parquet,
        )
        elapsed = time.perf_counter() - t0
    finally:
        tmp_parquet.unlink(missing_ok=True)

    processed = summary["processed"]
    errors = summary["errors"]
    if processed == 0:
        print("ERROR: 0 images processed (all may already be cached or missing).")
        print("Tip: the throughput tool writes to a temp parquet to bypass the cache.")
        sys.exit(1)

    per_image_s = elapsed / processed
    projected_s = per_image_s * FULL_DATASET_SIZE
    projected_h = projected_s / 3600.0
    over_budget = projected_h > BUDGET_HOURS
    flag = "WARNING: OVER BUDGET" if over_budget else "OK"

    lines = [
        "# Phase 4 -- Pipeline Throughput Measurement",
        "",
        f"Date: {time.strftime('%Y-%m-%d')}",
        f"Variant: `{variant}`",
        "",
        "## Timing results",
        "",
        "| Metric | Value |",
        "|---|---|",
        f"| Sample size | {actual_n} images |",
        f"| Processed | {processed} |",
        f"| Errors | {errors} |",
        f"| Wall-clock elapsed | {elapsed:.1f} s |",
        f"| Per-image | {per_image_s:.3f} s |",
        f"| Images/sec | {1.0/per_image_s:.2f} |",
        "",
        "## 34k projection",
        "",
        "| Metric | Value |",
        "|---|---|",
        f"| Full dataset size | {FULL_DATASET_SIZE:,} images |",
        f"| Projected wall-clock | {projected_h:.2f} hours ({projected_s/60:.0f} min) |",
        f"| Budget | {BUDGET_HOURS} hours |",
        f"| Status | **{flag}** |",
        "",
    ]

    if over_budget:
        headroom = projected_h - BUDGET_HOURS
        lines += [
            "## Action required",
            "",
            f"Projected run ({projected_h:.2f} h) exceeds budget by {headroom:.2f} h. "
            "Optimize before launching full run. Candidates:",
            "",
            "- Increase `DETECT_BATCH_SIZE` / `SEGMENT_BATCH_SIZE` in `config.py`",
            "- Pre-filter images with no filesystem file (skip DB rows where the image is missing)",
            "- Profile with `py-spy` to identify hotspot (image decode vs. DINO inference vs. SAM)",
            "",
        ]
    else:
        lines += [
            "## Decision",
            "",
            f"Projected run ({projected_h:.2f} h) is within the {BUDGET_HOURS} h budget. "
            "Proceed to T1 gate sweep.",
            "",
        ]

    report = "\n".join(lines)
    out_path = ROOT / args.out
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(report)
    print(report)
    print(f"\n[throughput] report written to {out_path}")
    if over_budget:
        sys.exit(2)  # non-zero so CI can catch it


if __name__ == "__main__":
    main()
```

**Purpose:** Sweep ML label probability thresholds (0.01-0.99) to find per-label thresholds that hit >= 0.94 gate precision against the human-labeled subset. Outputs a markdown table + recommended thresholds.

**Pre-condition:** Phase 3 ML labelers must have written probability columns to the parquet (e.g., `ml_prob_blur_unusable`, `ml_prob_blur_usable`). If these columns don't exist, the sweep runs in rule-only mode and notes that ML thresholds are not applicable.

- [ ] **Step 1: Create `tools/gate_sweep.py`**

```python
#!/usr/bin/env python3
"""Sweep ML-label probability thresholds to calibrate the drawability gate.

For each ML label (mask_blur_unusable, mask_blur_usable, mask_poor-contrast),
sweeps probability thresholds 0.01-0.99 in 0.05 steps. At each threshold
combo, re-derives the gate decision for every labeled image and computes
precision / recall / F1 vs. human gate decisions.

Human gate decision = KEEP iff all four columns at their "good" default
(matching gate.py:decide_drawability logic).

Outputs:
  docs/gate_sweep_<date>.md   -- full table + recommendations per ML label
  stdout                      -- summary of recommended thresholds

Usage:
    .venv/bin/python tools/gate_sweep.py [--step 0.05] [--precision-target 0.94]
"""
from __future__ import annotations
import argparse
import json
import sys
import time
from dataclasses import dataclass, field
from itertools import product
from pathlib import Path
from typing import Optional

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))

import polars as pl

from scripts.detect_subjects.config import (
    CACHE_DIR,
    PARQUET_PATH,
)
from scripts.detect_subjects import config as cfg
from scripts.detect_subjects.gate import GateDecision

PRECISION_TARGET = 0.94

# ML probability columns written by Phase 3 labelers.
# If a column is absent from the parquet, its threshold is skipped.
ML_PROB_COLS = {
    "mask_blur_unusable": "ml_prob_blur_unusable",
    "mask_blur_usable":   "ml_prob_blur_usable",
    "mask_poor-contrast": "ml_prob_poor_contrast",
}

_BBOX_GOOD = "bbox_correct-subject_not-clipped"
_BBOX_CONTENT_SINGLE = "bbox-content_single"


@dataclass(slots=True)
class SweepPoint:
    thresholds: dict[str, float]
    precision: float
    recall: float
    f1: float
    n_labeled: int
    n_keep_predicted: int
    n_keep_human: int


def _load_labels() -> dict[str, dict]:
    labels_path = CACHE_DIR / "labels.json"
    if not labels_path.exists():
        return {}
    return json.loads(labels_path.read_text() or "{}")


def _human_gate_decision(rec: dict) -> Optional[GateDecision]:
    """Derive human gate decision from a labels.json record.

    Returns None if the record is incomplete or unsure.
    KEEP iff all four columns at their good default.
    """
    if not rec or rec.get("unsure"):
        return None

    bbox = rec.get("bbox")
    if bbox is None:
        return None
    bbox_content_count = rec.get("bbox_content_count")
    bbox_too_small = bool(rec.get("bbox_too_small", False))
    mask_labels = rec.get("mask_labels") or []
    ml_labels = rec.get("ml_labels") or []

    label_record = {
        "bbox": bbox,
        "bbox_content_count": bbox_content_count or "",
        "bbox_too_small": bbox_too_small,
        "mask_labels": mask_labels,
        "ml_labels": ml_labels,
    }
    from scripts.detect_subjects.gate import decide_drawability
    return decide_drawability(label_record)


def _gate_decision_with_thresholds(
    row: dict,
    labels_rec: dict,
    ml_thresholds: dict[str, float],
    parquet_cols: set[str],
) -> Optional[GateDecision]:
    """Compute gate decision for a parquet row using given ML thresholds.

    Uses rule-based columns 1-2 from labels.json, then applies ML thresholds
    to columns 3-4 ML signals from the parquet probability columns.
    """
    if not labels_rec or labels_rec.get("unsure"):
        return None

    bbox = labels_rec.get("bbox")
    if bbox is None:
        return None
    if bbox != _BBOX_GOOD:
        return GateDecision.REJECT

    bbox_content_count = labels_rec.get("bbox_content_count") or ""
    if bbox_content_count != _BBOX_CONTENT_SINGLE:
        return GateDecision.REJECT
    if labels_rec.get("bbox_too_small", False):
        return GateDecision.REJECT

    blur_unusable_col = ML_PROB_COLS["mask_blur_unusable"]
    blur_usable_col = ML_PROB_COLS["mask_blur_usable"]

    if blur_unusable_col in parquet_cols:
        prob = row.get(blur_unusable_col) or 0.0
        t = ml_thresholds.get("mask_blur_unusable", 0.5)
        if prob >= t:
            return GateDecision.REJECT

    if blur_usable_col in parquet_cols:
        prob = row.get(blur_usable_col) or 0.0
        t = ml_thresholds.get("mask_blur_usable", 0.5)
        if prob >= t:
            return GateDecision.REJECT

    poor_contrast_col = ML_PROB_COLS["mask_poor-contrast"]
    if poor_contrast_col in parquet_cols:
        prob = row.get(poor_contrast_col) or 0.0
        t = ml_thresholds.get("mask_poor-contrast", 0.5)
        if prob >= t:
            return GateDecision.REJECT

    return GateDecision.KEEP


def _compute_metrics(
    parquet_rows: list[dict],
    labels: dict[str, dict],
    ml_thresholds: dict[str, float],
    parquet_cols: set[str],
) -> Optional[SweepPoint]:
    """Compute precision/recall/F1 for one threshold combination."""
    tp = fp = fn = tn = 0
    n_keep_human = 0
    n_keep_pred = 0

    for row in parquet_rows:
        image_id = row["image_id"]
        rec = labels.get(image_id)
        human = _human_gate_decision(rec)
        if human is None:
            continue
        predicted = _gate_decision_with_thresholds(row, rec, ml_thresholds, parquet_cols)
        if predicted is None:
            continue

        if human == GateDecision.KEEP:
            n_keep_human += 1
        if predicted == GateDecision.KEEP:
            n_keep_pred += 1

        if predicted == GateDecision.KEEP and human == GateDecision.KEEP:
            tp += 1
        elif predicted == GateDecision.KEEP and human == GateDecision.REJECT:
            fp += 1
        elif predicted == GateDecision.REJECT and human == GateDecision.KEEP:
            fn += 1
        else:
            tn += 1

    n = tp + fp + fn + tn
    if n == 0:
        return None

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

    return SweepPoint(
        thresholds=dict(ml_thresholds),
        precision=precision,
        recall=recall,
        f1=f1,
        n_labeled=n,
        n_keep_predicted=n_keep_pred,
        n_keep_human=n_keep_human,
    )


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--step", type=float, default=0.05,
                    help="Threshold step size (default: 0.05)")
    ap.add_argument("--precision-target", type=float, default=PRECISION_TARGET,
                    help="Minimum precision to qualify as a recommendation (default: 0.94)")
    ap.add_argument("--out", default=None,
                    help="Output markdown path (default: docs/gate_sweep_<date>.md)")
    args = ap.parse_args()

    date_str = time.strftime("%Y-%m-%d")
    out_path = ROOT / (args.out or f"docs/gate_sweep_{date_str}.md")

    variant = cfg.variant_tag()
    labels = _load_labels()
    if not labels:
        print("ERROR: no labels found in data/cache/labels.json -- run validator first.")
        sys.exit(1)

    if not PARQUET_PATH.exists():
        print(f"ERROR: parquet not found: {PARQUET_PATH}")
        sys.exit(1)

    df = pl.read_parquet(PARQUET_PATH).filter(pl.col("variant") == variant)
    if df.height == 0:
        print(f"ERROR: no rows for variant={variant}")
        sys.exit(1)

    parquet_cols = set(df.columns)
    parquet_rows = df.to_dicts()
    print(f"[gate_sweep] {df.height} parquet rows, {len(labels)} label records, variant={variant}")

    active_ml_labels = {
        label: col for label, col in ML_PROB_COLS.items()
        if col in parquet_cols
    }
    if not active_ml_labels:
        print("WARNING: no ML probability columns found in parquet.")
        print("Running in rule-only mode (single threshold combo at defaults).")

    thresholds_grid = [round(i * args.step, 3) for i in range(1, int(1.0 / args.step))]

    if active_ml_labels:
        label_names = list(active_ml_labels.keys())
        combos = [
            dict(zip(label_names, vals))
            for vals in product(thresholds_grid, repeat=len(label_names))
        ]
    else:
        combos = [{}]

    print(f"[gate_sweep] sweeping {len(combos)} threshold combinations...")

    results: list[SweepPoint] = []
    for i, thresholds in enumerate(combos):
        pt = _compute_metrics(parquet_rows, labels, thresholds, parquet_cols)
        if pt is not None:
            results.append(pt)
        if (i + 1) % 500 == 0:
            print(f"  [{i+1}/{len(combos)}]")

    if not results:
        print("ERROR: no labeled images contributed to sweep. Check labels.json format.")
        sys.exit(1)

    results.sort(key=lambda p: (-p.precision, -p.recall))

    meeting_target = [p for p in results if p.precision >= args.precision_target]
    best_per_label: dict[str, SweepPoint] = {}
    if meeting_target:
        for label_name in (active_ml_labels or {""}).keys():
            by_t: dict[float, list[SweepPoint]] = {}
            for pt in meeting_target:
                t = pt.thresholds.get(label_name, 0.5)
                by_t.setdefault(t, []).append(pt)
            best_t = max(by_t, key=lambda t: max(p.recall for p in by_t[t]))
            best_pt = max(by_t[best_t], key=lambda p: p.recall)
            best_per_label[label_name] = best_pt

    lines = [
        f"# Gate Sweep -- {date_str}",
        "",
        f"Variant: `{variant}`",
        f"Precision target: {args.precision_target:.2f}",
        f"Threshold step: {args.step}",
        f"Labeled images: {results[0].n_labeled if results else 0}",
        f"Combinations swept: {len(combos)}",
        "",
        "## Results (top 30 by precision, then recall)",
        "",
    ]
    if active_ml_labels:
        threshold_cols = "".join(f" | `{n}` threshold" for n in active_ml_labels)
    else:
        threshold_cols = ""

    lines.append(
        f"| precision | recall | F1 | keeps(pred) | keeps(human){threshold_cols} |"
    )
    sep = "|---|---|---|---|---" + "|---" * len(active_ml_labels) + "|"
    lines.append(sep)

    for pt in results[:30]:
        t_cols = "".join(f" | {pt.thresholds.get(n, '-'):.2f}" for n in (active_ml_labels or {}))
        lines.append(
            f"| {pt.precision:.3f} | {pt.recall:.3f} | {pt.f1:.3f} "
            f"| {pt.n_keep_predicted} | {pt.n_keep_human}{t_cols} |"
        )

    lines += ["", "## Recommended thresholds", ""]
    if meeting_target:
        lines.append(
            f"The following thresholds meet the >= {args.precision_target:.2f} "
            "precision target with maximum recall:\n"
        )
        for label_name, pt in best_per_label.items():
            t = pt.thresholds.get(label_name, 0.5)
            lines.append(
                f"- **`{label_name}`**: threshold = `{t:.2f}` "
                f"(precision={pt.precision:.3f}, recall={pt.recall:.3f}, F1={pt.f1:.3f})"
            )
        lines += [
            "",
            "Copy these into `config.py` under `# --- Gate thresholds ---`.",
        ]
    else:
        lines += [
            f"WARNING: No threshold combination achieved precision >= {args.precision_target:.2f}.",
            "",
            "Best achievable precision:",
            f"  {results[0].precision:.3f} (recall={results[0].recall:.3f})",
            "",
            "Action: re-enter Phase 3 active learning to grow the labeled set, "
            "then re-run this sweep.",
        ]

    report = "\n".join(lines) + "\n"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(report)
    print(report)
    print(f"\n[gate_sweep] report written to {out_path}")

    if not meeting_target:
        sys.exit(2)


if __name__ == "__main__":
    main()
```

```python
"""Drawability gate -- combines all label sources into one keep/reject decision.

Per the strict gate definition (modular pipeline design spec, 2026-05-15):
  Reject if ANY of:
    - S1 not bbox_correct-subject_not-clipped
    - S2 count != bbox-content_single, OR bbox-content_subject-too-small set
    - S3 any selection other than mask_good (including soft-reject _usable variants),
         OR ML blur probability >= GATE_THRESHOLD_BLUR_UNUSABLE / _USABLE
    - S4 any selection other than ml_good,
         OR ML poor-contrast probability >= GATE_THRESHOLD_POOR_CONTRAST

  Keep otherwise. bbox-content_image-multi-bug is informational and does NOT
  contribute to the gate decision.
"""
from __future__ import annotations
from enum import Enum
from typing import Optional


class GateDecision(Enum):
    KEEP = "keep"
    REJECT = "reject"


_BBOX_GOOD = "bbox_correct-subject_not-clipped"
_BBOX_CONTENT_SINGLE = "bbox-content_single"


def decide_drawability(
    label_record: dict,
    *,
    blur_unusable_prob: Optional[float] = None,
    blur_usable_prob: Optional[float] = None,
    poor_contrast_prob: Optional[float] = None,
) -> GateDecision:
    """Return KEEP if all four columns at their 'good' default, REJECT otherwise.

    Required label_record keys (KeyError on missing -- safe-fail):
      bbox: str                       # S1 -- one of bbox_*
      bbox_content_count: str         # S2 count -- one of bbox-content_*
      bbox_too_small: bool            # S2 independent flag
      mask_labels: list[str]          # S3 selections; empty list = mask_good
      ml_labels: list[str]            # S4 selections; empty list = ml_good

    Optional keyword arguments (ML probability scores from Phase 3 labelers):
      blur_unusable_prob: float | None   -- fires REJECT if >= GATE_THRESHOLD_BLUR_UNUSABLE
      blur_usable_prob: float | None     -- fires REJECT if >= GATE_THRESHOLD_BLUR_USABLE
      poor_contrast_prob: float | None   -- fires REJECT if >= GATE_THRESHOLD_POOR_CONTRAST

    Optional label_record keys:
      bbox_content_image_multi_bug: bool  # S2 informational -- NOT a gate signal

    mask_good and ml_good are represented as empty lists in mask_labels / ml_labels
    respectively, NOT as list entries. A non-empty list means at least one non-good
    selection was made, which triggers REJECT.
    """
    from scripts.detect_subjects.config import (
        GATE_THRESHOLD_BLUR_UNUSABLE,
        GATE_THRESHOLD_BLUR_USABLE,
        GATE_THRESHOLD_POOR_CONTRAST,
    )

    if label_record["bbox"] != _BBOX_GOOD:
        return GateDecision.REJECT
    if label_record["bbox_content_count"] != _BBOX_CONTENT_SINGLE:
        return GateDecision.REJECT
    if label_record["bbox_too_small"]:
        return GateDecision.REJECT
    if label_record["mask_labels"]:
        return GateDecision.REJECT

    # ML-derived signals for column S3 (mask quality).
    if blur_unusable_prob is not None and blur_unusable_prob >= GATE_THRESHOLD_BLUR_UNUSABLE:
        return GateDecision.REJECT
    if blur_usable_prob is not None and blur_usable_prob >= GATE_THRESHOLD_BLUR_USABLE:
        return GateDecision.REJECT

    if label_record["ml_labels"]:
        return GateDecision.REJECT

    # ML-derived signals for column S4 (image quality).
    if poor_contrast_prob is not None and poor_contrast_prob >= GATE_THRESHOLD_POOR_CONTRAST:
        return GateDecision.REJECT

    # bbox_content_image_multi_bug is informational; not checked.
    return GateDecision.KEEP
```

```python
#!/usr/bin/env python3
"""Drive the full 34k pipeline run with checkpoint/resume.

Reads the full image list from data/db/line-of-bugs.db, skips images already
processed for the current variant, processes in batches of BATCH_SIZE, and
writes per-batch checkpoint JSON to data/cache/full_run_progress.json.
Re-running resumes from the last checkpoint.

Usage:
    .venv/bin/python tools/run_full_dataset.py [--batch-size 500] [--dry-run]
"""
from __future__ import annotations
import argparse
import json
import sqlite3
import sys
import time
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))

from scripts.detect_subjects.config import (
    CACHE_DIR,
    DATA_DIR,
    PARQUET_PATH,
)
from scripts.detect_subjects import config as cfg
from scripts.detect_subjects.caches import load_completed_pairs
from scripts.detect_subjects.classify import run_v1_on_sample

CHECKPOINT_PATH = CACHE_DIR / "full_run_progress.json"
VALID_SOURCES = {"inaturalist", "bugwood"}


def _load_all_images(db_path: Path) -> list[dict]:
    """Return all image rows with a filename from the DB, active sources only."""
    con = sqlite3.connect(str(db_path))
    try:
        rows = con.execute(
            "SELECT image_id, filename, source "
            "FROM images "
            "WHERE filename IS NOT NULL AND source IN ('inaturalist', 'bugwood')"
        ).fetchall()
    finally:
        con.close()
    return [{"image_id": r[0], "filename": r[1], "source": r[2]} for r in rows]


def _load_checkpoint() -> dict:
    if not CHECKPOINT_PATH.exists():
        return {"batches": [], "total_processed": 0, "total_errors": 0}
    try:
        return json.loads(CHECKPOINT_PATH.read_text())
    except Exception:
        return {"batches": [], "total_processed": 0, "total_errors": 0}


def _save_checkpoint(state: dict) -> None:
    CHECKPOINT_PATH.parent.mkdir(parents=True, exist_ok=True)
    tmp = CHECKPOINT_PATH.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(state, indent=2))
    tmp.replace(CHECKPOINT_PATH)


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--batch-size", type=int, default=500,
                    help="Images per batch (default: 500)")
    ap.add_argument("--dry-run", action="store_true",
                    help="Print plan (images to process, batches) without running")
    ap.add_argument("--reset", action="store_true",
                    help="Delete checkpoint and restart from scratch")
    args = ap.parse_args()

    db_path = DATA_DIR / "db" / "line-of-bugs.db"
    if not db_path.exists():
        print(f"ERROR: DB not found: {db_path}")
        sys.exit(1)

    variant = cfg.variant_tag()
    print(f"[full_run] variant={variant}")

    if args.reset and CHECKPOINT_PATH.exists():
        CHECKPOINT_PATH.unlink()
        print("[full_run] checkpoint reset.")

    all_rows = _load_all_images(db_path)
    print(f"[full_run] {len(all_rows)} images in DB (sources: inaturalist, bugwood)")

    completed = load_completed_pairs(PARQUET_PATH)
    to_process = [r for r in all_rows if (r["image_id"], variant) not in completed]
    print(f"[full_run] {len(completed)} already in parquet, {len(to_process)} to process")

    if args.dry_run:
        n_batches = (len(to_process) + args.batch_size - 1) // args.batch_size
        print(f"[dry-run] would process {len(to_process)} images in {n_batches} batches "
              f"of {args.batch_size}")
        return

    if not to_process:
        print("[full_run] all images already processed. Done.")
        return

    checkpoint = _load_checkpoint()
    done_ids: set[str] = set()
    for batch_entry in checkpoint["batches"]:
        done_ids.update(batch_entry.get("image_ids_done") or [])

    to_process = [r for r in to_process if r["image_id"] not in done_ids]
    print(f"[full_run] {len(done_ids)} checkpointed, {len(to_process)} remaining")

    n_batches = (len(to_process) + args.batch_size - 1) // args.batch_size
    t_global_start = time.perf_counter()
    total_processed = checkpoint.get("total_processed", 0)
    total_errors = checkpoint.get("total_errors", 0)

    for batch_idx in range(n_batches):
        start = batch_idx * args.batch_size
        batch = to_process[start: start + args.batch_size]
        print(f"\n[full_run] batch {batch_idx+1}/{n_batches}: {len(batch)} images")

        t_batch_start = time.perf_counter()
        try:
            summary = run_v1_on_sample(batch, gt_index=None, parquet_path=PARQUET_PATH)
        except Exception as e:
            print(f"[full_run] FATAL ERROR in batch {batch_idx}: {e}")
            raise

        batch_elapsed = time.perf_counter() - t_batch_start
        total_processed += summary["processed"]
        total_errors += summary["errors"]

        batch_entry = {
            "batch_idx": batch_idx,
            "image_ids_done": [r["image_id"] for r in batch],
            "image_ids_failed": [],
            "batch_elapsed_s": round(batch_elapsed, 2),
            "processed": summary["processed"],
            "errors": summary["errors"],
        }
        checkpoint["batches"].append(batch_entry)
        checkpoint["total_processed"] = total_processed
        checkpoint["total_errors"] = total_errors
        _save_checkpoint(checkpoint)

        elapsed_total = time.perf_counter() - t_global_start
        rate = total_processed / elapsed_total if elapsed_total > 0 else 0
        remaining = len(to_process) - (batch_idx + 1) * args.batch_size
        eta_s = remaining / rate if rate > 0 else 0
        print(f"[full_run] cumulative: processed={total_processed} errors={total_errors} "
              f"rate={rate:.2f}/s ETA={eta_s/3600:.2f}h")

    total_elapsed = time.perf_counter() - t_global_start
    print(f"\n[full_run] DONE: processed={total_processed} errors={total_errors} "
          f"total_hours={total_elapsed/3600:.2f}")


if __name__ == "__main__":
    main()
```

```python
    The rendered HTML includes _sample_review: true in the DATA records so the
    JS can switch to agree/disagree mode.

    if not parquet_path.exists():
        raise FileNotFoundError(f"parquet not found: {parquet_path}")

    df = pl.read_parquet(parquet_path)

    if "gate_decision" in df.columns:
        keeps = df.filter(pl.col("gate_decision") == "keep")
        rejects = df.filter(pl.col("gate_decision") == "reject")
    else:
        keeps = df.filter(pl.col("framing_quality").is_in(["good", "tight"]))
        rejects = df.filter(~pl.col("framing_quality").is_in(["good", "tight"]))

            f"Not enough rows: {keeps.height} keeps, {rejects.height} rejects "
            f"(need {n_each} each). Run the full dataset pipeline first."

    sample_rejects = rejects.sample(n_each, seed=seed)
    sample_df = pl.concat([sample_keeps, sample_rejects]).sample(fraction=1.0, seed=seed)

    db_index = _load_db_index()

            "bbox_area_ratio": row["bbox_area_ratio"],
            "bbox_long_edge_px": row.get("bbox_long_edge_px"),
            "confidence": row["confidence"],
            "lab_delta_e": row["lab_delta_e"],
            "offcenter": row["offcenter"],
```

- [ ] **Step 1: Create `tools/analyze_sample_review.py`**

```python
#!/usr/bin/env python3
"""Analyze sample-review results and produce a ship/iterate decision.

Reads data/cache/labels.json (key: __sample_review__) after the user has
clicked through 100 sample images in sample-review mode.

Decision rules (parent spec para 527):
  <= 3 disagreements out of 100  -> SHIP (proceed to Phase H)
  4-10 disagreements             -> ITERATE (analyze by label class)
  >= 11 disagreements            -> SIGNIFICANT MISS (revisit gate or Phase 2)

For ITERATE: identifies which label class drives the most disagreements.
If one label accounts for >= 50% of disagreement images, recommends Phase 3
active learning for that label. Otherwise recommends gate threshold re-tune.

Usage:
    .venv/bin/python tools/analyze_sample_review.py [--labels path] [--out path]
"""
from __future__ import annotations
import argparse
import json
import sys
import time
from pathlib import Path
from collections import Counter
from typing import Optional

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))

from scripts.detect_subjects.config import CACHE_DIR, PARQUET_PATH

SHIP_THRESHOLD = 3
SIGNIFICANT_MISS_THRESHOLD = 11
DOMINANT_CLASS_FRACTION = 0.50


def _load_results(labels_path: Path) -> tuple[list[dict], dict]:
    """Return (per_image_results, summary_meta) from labels.json."""
    if not labels_path.exists():
        raise FileNotFoundError(f"labels.json not found: {labels_path}")
    data = json.loads(labels_path.read_text() or "{}")
    sr = data.get("__sample_review__")
    if sr is None:
        raise KeyError("No __sample_review__ key in labels.json. "
                       "Complete the sample-review UI session first.")
    return sr["results"], sr


def _dominant_label_class(
    disagreement_image_ids: list[str],
    parquet_path: Path,
) -> tuple[Optional[str], float]:
    """Find the label class driving the most disagreements.

    Returns (dominant_label, fraction) or (None, 0.0) if parquet is missing
    or no suggested_labels data is available.
    """
    if not parquet_path.exists() or not disagreement_image_ids:
        return None, 0.0

    import polars as pl
    df = pl.read_parquet(parquet_path).filter(
        pl.col("image_id").is_in(disagreement_image_ids)
    )
    if df.height == 0:
        return None, 0.0

    label_counts: Counter = Counter()
    for row in df.iter_rows(named=True):
        for label in (row.get("suggested_labels") or []):
            label_counts[label] += 1

    if not label_counts:
        return None, 0.0

    total = sum(label_counts.values())
    top_label, top_count = label_counts.most_common(1)[0]
    fraction = top_count / total if total > 0 else 0.0
    return top_label, fraction


def _build_recommendation(
    n_total: int,
    n_disagreements: int,
    n_reviewed: int,
    dominant_label: Optional[str],
    dominant_fraction: float,
) -> str:
    """Return a concrete recommendation string based on disagreement count and breakdown."""
    if n_disagreements <= SHIP_THRESHOLD:
        return (
            f"SHIP: {n_disagreements}/{n_reviewed} disagreements "
            f"(<= {SHIP_THRESHOLD} threshold). "
            "Gate precision criterion met. Proceed to Phase H (gallery wiring)."
        )

    if n_disagreements >= SIGNIFICANT_MISS_THRESHOLD:
        return (
            f"SIGNIFICANT MISS: {n_disagreements}/{n_reviewed} disagreements "
            f"(>= {SIGNIFICANT_MISS_THRESHOLD} threshold). "
            "Action: revisit gate thresholds (rerun gate_sweep.py at a lower precision "
            "target) OR escalate to Phase 2 (SAM 3 bbox quality review). "
            "Do NOT ship until disagreements drop to <= 3."
        )

    # 4-10 disagreements -- ITERATE path.
    if dominant_label is not None and dominant_fraction >= DOMINANT_CLASS_FRACTION:
        return (
            f"ITERATE -- label class dominant: {n_disagreements}/{n_reviewed} disagreements; "
            f"{dominant_fraction*100:.0f}% driven by label '{dominant_label}'. "
            f"Action: re-enter Phase 3 active learning for '{dominant_label}' -- "
            "grow its labeled set past 100, retrain classifier, re-run gate_sweep.py, "
            "then re-run the full 34k pipeline and repeat sample review."
        )

    dominant_note = (
        f"top class: '{dominant_label}' at {dominant_fraction*100:.0f}%"
        if dominant_label else "no label data"
    )
    return (
        f"ITERATE -- disagreements spread: {n_disagreements}/{n_reviewed} disagreements "
        f"with no dominant label class ({dominant_note}). "
        "Action: re-tune gate thresholds (rerun gate_sweep.py, possibly accept a lower "
        "precision target to improve recall) and re-run sample review. "
        "If no improvement after one iteration, escalate to Phase 2."
    )


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--labels", default=str(CACHE_DIR / "labels.json"),
                    help="Path to labels.json (default: data/cache/labels.json)")
    ap.add_argument("--out", default=None,
                    help="Output markdown path (default: docs/phase4_sample_review_<date>.md)")
    args = ap.parse_args()

    labels_path = Path(args.labels)
    date_str = time.strftime("%Y-%m-%d")
    out_path = ROOT / (args.out or f"docs/phase4_sample_review_{date_str}.md")

    try:
        results, meta = _load_results(labels_path)
    except (FileNotFoundError, KeyError) as e:
        print(f"ERROR: {e}")
        sys.exit(1)

    n_reviewed = len(results)
    disagreements = [r for r in results if not r.get("user_agrees")]
    n_disagreements = len(disagreements)
    n_agrees = n_reviewed - n_disagreements

    disagree_by_gate: Counter = Counter()
    disagree_image_ids = []
    for r in disagreements:
        disagree_by_gate[r.get("gate_decision", "unknown")] += 1
        disagree_image_ids.append(r["image_id"])

    dominant_label, dominant_fraction = _dominant_label_class(
        disagree_image_ids, PARQUET_PATH
    )

    recommendation = _build_recommendation(
        n_total=meta.get("total", n_reviewed),
        n_disagreements=n_disagreements,
        n_reviewed=n_reviewed,
        dominant_label=dominant_label,
        dominant_fraction=dominant_fraction,
    )

    lines = [
        f"# Phase 4 Sample Review -- {date_str}",
        "",
        "## Results",
        "",
        "| Metric | Value |",
        "|---|---|",
        f"| Total sampled | {meta.get('total', n_reviewed)} |",
        f"| Reviewed | {n_reviewed} |",
        f"| Agrees | {n_agrees} |",
        f"| Disagreements | {n_disagreements} |",
        f"| Disagreement rate | {n_disagreements/max(n_reviewed,1)*100:.1f}% |",
        "",
        "## Disagreement breakdown by gate decision",
        "",
        "| Gate decision | Disagreements |",
        "|---|---|",
    ]
    for gd, count in disagree_by_gate.most_common():
        lines.append(f"| {gd} | {count} |")

    if dominant_label:
        lines += [
            "",
            f"**Dominant label class:** `{dominant_label}` "
            f"({dominant_fraction*100:.0f}% of disagreement images had this label)",
        ]

    lines += [
        "",
        "## Decision",
        "",
        recommendation,
        "",
    ]

    report = "\n".join(lines) + "\n"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(report)
    print(report)
    print(f"\n[analyze] report written to {out_path}")

    # Exit codes for scripting: 0=ship, 1=iterate, 2=significant miss.
    if n_disagreements <= SHIP_THRESHOLD:
        sys.exit(0)
    elif n_disagreements >= SIGNIFICANT_MISS_THRESHOLD:
        sys.exit(2)
    else:
        sys.exit(1)


if __name__ == "__main__":
    main()
```

**Purpose:** If parquet size exceeds 10 GB after the 34k run, prune rows for variants no longer in active use (e.g., legacy `v1_dino_insectsam` rows once SAM 3 variant is committed).

- [ ] **Step 1: Create `tools/prune_legacy_variants.py`**

```python
#!/usr/bin/env python3
"""Prune legacy variant rows from the framing detections parquet.

A "legacy" variant is any variant string that no longer matches
cfg.variant_tag() (the currently configured detector+segmenter combo).

Usage:
    .venv/bin/python tools/prune_legacy_variants.py           # dry-run (default)
    .venv/bin/python tools/prune_legacy_variants.py --apply   # actually prune
    .venv/bin/python tools/prune_legacy_variants.py --force   # prune under size threshold
"""
from __future__ import annotations
import argparse
import sys
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(ROOT))

import polars as pl

from scripts.detect_subjects.config import PARQUET_PATH
from scripts.detect_subjects import config as cfg

SIZE_THRESHOLD_GB = 10.0


def _parquet_size_gb(path: Path) -> float:
    return path.stat().st_size / 1e9 if path.exists() else 0.0


def main() -> None:
    ap = argparse.ArgumentParser()
    ap.add_argument("--apply", action="store_true",
                    help="Actually prune (default is dry-run)")
    ap.add_argument("--force", action="store_true",
                    help="Prune even if parquet is under the size threshold")
    args = ap.parse_args()

    if not PARQUET_PATH.exists():
        print(f"ERROR: parquet not found: {PARQUET_PATH}")
        sys.exit(1)

    size_gb = _parquet_size_gb(PARQUET_PATH)
    active_variant = cfg.variant_tag()
    print(f"[prune] parquet: {PARQUET_PATH}")
    print(f"[prune] size: {size_gb:.2f} GB (threshold: {SIZE_THRESHOLD_GB} GB)")
    print(f"[prune] active variant: {active_variant}")

    if size_gb < SIZE_THRESHOLD_GB and not args.force:
        print(f"[prune] parquet is under {SIZE_THRESHOLD_GB} GB -- no pruning needed.")
        print("       Use --force to prune anyway.")
        return

    df = pl.read_parquet(PARQUET_PATH)
    all_variants = df["variant"].unique().to_list()
    legacy_variants = [v for v in all_variants if v != active_variant]

    print(f"[prune] variants in parquet: {sorted(all_variants)}")
    print(f"[prune] legacy variants to remove: {sorted(legacy_variants)}")

    if not legacy_variants:
        print("[prune] no legacy variants found -- nothing to remove.")
        return

    active_df = df.filter(pl.col("variant") == active_variant)
    legacy_rows = df.height - active_df.height

    print(f"[prune] rows: {df.height} total, {active_df.height} active, "
          f"{legacy_rows} legacy (to remove)")

    if args.apply:
        print("[prune] writing pruned parquet...")
        active_df.write_parquet(PARQUET_PATH, compression="snappy")
        new_size = _parquet_size_gb(PARQUET_PATH)
        delta_gb = size_gb - new_size
        print(f"[prune] done: {new_size:.2f} GB (removed {delta_gb:.2f} GB, "
              f"{legacy_rows} rows)")
    else:
        projected_new_gb = size_gb * (active_df.height / df.height) if df.height else 0.0
        print(f"[prune] DRY RUN -- would remove {legacy_rows} rows "
              f"(~{size_gb - projected_new_gb:.2f} GB savings). "
              "Run with --apply to execute.")


if __name__ == "__main__":
    main()
```
